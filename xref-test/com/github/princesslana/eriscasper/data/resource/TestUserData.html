3   import com.github.princesslana.eriscasper.data.DataAssert;
4   import com.github.princesslana.eriscasper.data.Snowflake;
5   import java.util.Optional;
6   import org.testng.annotations.Test;
7   
8   public class TestUserData {
9   
10    @Test
11    public void deserialize_whenValidPayload_shouldDeserialize() {
12      String payload =
13          "{     \"username\"     :\"LaBotuel\","
14              + "\"id\"           :\"417388135027048495\","
15              + "\"discriminator\":\"7013\","
16              + "\"bot\"          :true,"
17              + "\"avatar\"       :null"
18              + "}";
19  
20      DataAssert.thatFromJson(payload, User.class)
21          .hasFieldOrPropertyWithValue("id", Snowflake.of("417388135027048495"))
22          .hasFieldOrPropertyWithValue("username", "LaBotuel")
23          .hasFieldOrPropertyWithValue("discriminator", "7013")
24          .hasFieldOrPropertyWithValue("avatar", Optional.empty())
25          .hasFieldOrPropertyWithValue("bot", Optional.of(true))
26          .hasFieldOrPropertyWithValue("mfaEnabled", Optional.empty())
27          .hasFieldOrPropertyWithValue("verified", Optional.empty())
28          .hasFieldOrPropertyWithValue("email", Optional.empty());
29    }
30  
31    /**
32     * @see <a href="https://discordapp.com/developers/docs/resources/user#user-object-example-user">
33     *     https://discordapp.com/developers/docs/resources/user#user-object-example-user</a>
34     */
35    @Test
36    public void deserialize_whenExamplePayload_shouldDeseralize() {
37      String payload =
38          "{     \"id\"           : \"80351110224678912\","
39              + "\"username\"     : \"Nelly\","
40              + "\"discriminator\": \"1337\","
41              + "\"avatar\"       : \"8342729096ea3675442027381ff50dfe\","
42              + "\"verified\"     : true,"
43              + "\"email\"        : \"nelly@discordapp.com\""
44              + "}";
45  
46      DataAssert.thatFromJson(payload, User.class)
47          .hasFieldOrPropertyWithValue("id", Snowflake.of("80351110224678912"))
48          .hasFieldOrPropertyWithValue("username", "Nelly")
49          .hasFieldOrPropertyWithValue("discriminator", "1337")
50          .hasFieldOrPropertyWithValue("avatar", Optional.of("8342729096ea3675442027381ff50dfe"))
51          .hasFieldOrPropertyWithValue("bot", Optional.empty())
52          .hasFieldOrPropertyWithValue("mfaEnabled", Optional.empty())
53          .hasFieldOrPropertyWithValue("verified", Optional.of(true))
54          .hasFieldOrPropertyWithValue("email", Optional.of("nelly@discordapp.com"));
55    }
