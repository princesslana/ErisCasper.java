25  public class TestPingAndEchoRobot {
26  
27    @Mock private Routes routes;
28  
29    private Bot subject;
30  
31    @BeforeMethod
32    public void subject() {
33      MockitoAnnotations.initMocks(this);
34      subject = PingAndEchoRobot.create();
35    }
36  
37    @Test
38    public void apply_whenPing_shouldPong() {
39      PublishSubject<Event> events = PublishSubject.create();
40  
41      BotContext ctx = new BotContext(events, routes, null);
42  
43      TestObserver<Void> subscriber = subject.apply(ctx).test();
44  
45      Snowflake channelId = DiscordFaker.snowflake();
46      User author = DataFaker.user();
47  
48      events.onNext(
49          MessageCreate.of(
50              ImmutableMessage.copyOf(DataFaker.message())
51                  .withAuthor(author)
52                  .withContent("+ping")
53                  .withChannelId(channelId)));
54  
55      String expectedResponse = Users.mentionByNickname(author) + " pong";
56  
57      then(routes)
58          .should()
59          .execute(
60              RouteCatalog.createMessage(channelId), SendMessageRequest.ofText(expectedResponse));
61  
62      subscriber.assertNotComplete();
63    }
64  }
