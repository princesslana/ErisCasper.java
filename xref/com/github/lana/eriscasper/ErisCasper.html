1   package com.github.lana.eriscasper;
2   
3   import io.reactivex.Observable;
4   import javax.security.auth.login.LoginException;
5   import net.dv8tion.jda.core.AccountType;
6   import net.dv8tion.jda.core.JDA;
7   import net.dv8tion.jda.core.JDABuilder;
8   import net.dv8tion.jda.core.events.Event;
9   import net.dv8tion.jda.core.hooks.EventListener;
10  
11  public class ErisCasper {
12  
13    private final Observable<Event> events;
14  
15    public ErisCasper(Observable<Event> events) {
16      this.events = events;
17    }
18  
19    public Observable<Event> events() {
20      return events;
21    }
22  
23    public static ErisCasper create() {
24      try {
25        return create(
26            new JDABuilder(AccountType.BOT).setToken(System.getenv("EC_TOKEN")).buildBlocking());
27      } catch (InterruptedException | LoginException e) {
28        throw new ErisCasperFatalException("Error building JDA instance", e);
29      }
30    }
31  
32    public static ErisCasper create(JDA jda) {
33      Observable<Event> events =
34          Observable.create(
35              emitter -> {
36                EventListener el = emitter::onNext;
37                jda.addEventListener(el);
38              });
39  
40      return new ErisCasper(events);
41    }
42  }
