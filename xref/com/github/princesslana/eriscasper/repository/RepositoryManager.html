3   import com.github.princesslana.eriscasper.data.event.Event;
4   import com.github.princesslana.eriscasper.repository.event.UsersFromEvents;
5   import com.google.common.base.Preconditions;
6   import io.reactivex.Observable;
7   import java.util.concurrent.ConcurrentHashMap;
8   import java.util.concurrent.ConcurrentMap;
9   
10  public class RepositoryManager {
11    private final ConcurrentMap<String, Object> repositories = new ConcurrentHashMap<>();
12  
13    private RepositoryManager() {}
14  
15    @SuppressWarnings("unchecked")
16    public <R> R get(RepositoryDefinition<R> def) {
17      Preconditions.checkState(
18          repositories.containsKey(def.getName()), "No such repository: %s", def.getName());
19  
20      return (R) repositories.get(def.getName());
21    }
22  
23    private <R> void put(RepositoryDefinition<R> def, R repository) {
24      Preconditions.checkState(
25          !repositories.containsKey(def.getName()),
26          "Duplicate repository created: %s",
27          def.getName());
28  
29      repositories.put(def.getName(), repository);
30    }
31  
32    public static RepositoryManager create(Observable<Event> events) {
33      RepositoryManager rm = new RepositoryManager();
34      rm.put(RepositoryDefinition.USER, new UsersFromEvents(events));
35      return rm;
36    }
37  }
