5   import com.github.princesslana.eriscasper.data.resource.User;
6   import com.github.princesslana.eriscasper.repository.UserRepository;
7   import io.reactivex.Observable;
8   import io.reactivex.Single;
9   import io.reactivex.observables.ConnectableObservable;
10  
11  public class UsersFromEvents implements UserRepository {
12  
13    private ConnectableObservable<User> self;
14  
15    public UsersFromEvents(Observable<Event> events) {
16      self = events.ofType(ReadyEvent.class).map(ReadyEvent::unwrap).map(d -> d.getUser()).replay(1);
17  
18      self.connect();
19    }
20  
21    public Single<User> getSelf() {
22      return self.firstOrError();
23    }
24  }
