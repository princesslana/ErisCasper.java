1   package com.github.princesslana.eriscasper.gateway;

3   import com.fasterxml.jackson.annotation.JsonCreator;
4   import com.fasterxml.jackson.annotation.JsonValue;
5   import java.util.Arrays;
6   
7   /**
8    * @see <a href="https://discordapp.com/developers/docs/topics/opcodes-and-status-codes#gateway">
9    *     https://discordapp.com/developers/docs/topics/opcodes-and-status-codes#gateway</a>
10   */
11  public enum OpCode {
12    DISPATCH(0),
13    HEARTBEAT(1),
14    IDENTIFY(2),
15    STATUS_UPDATE(3),
16    VOICE_STATE_UPDATE(4),
17    VOICE_SERVER_PING(5),
18    RESUME(6),
19    RECONNECT(7),
20    REQUEST_GUILD_MEMBERS(8),
21    INVALID_SESSION(9),
22    HELLO(10),
23    HEARTBEAT_ACK(11);
24  
25    private final int code;
26  
27    private OpCode(int code) {
28      this.code = code;
29    }
30  
31    @JsonValue
32    public int getCode() {
33      return code;
34    }
35  
36    @JsonCreator
37    public static OpCode fromCode(int code) {
38      return Arrays.stream(OpCode.values())
39          .filter(o -> o.code == code)
40          .findFirst()
41          .orElseThrow(() -> new IllegalStateException("Unknown opcode: " + code));
42    }
43  }
