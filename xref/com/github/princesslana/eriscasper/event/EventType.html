9   /**
10   * Possible event types that we can receive from Discord.
11   *
12   * <p>Each EventType is associated with an {@link EventFactory} so that it is known how to construct
13   * an {@link Event} from a {@link com.github.princesslana.eriscasper.gateway.Payload}.
14   *
15   * <p>It is tempting to try and merge EventType and {@link EventFactory} into a single class, but we
16   * can not match the dataClass and factory types using generics within the EventType enum.
17   *
18   * @see <a
19   *     href="https://discordapp.com/developers/docs/topics/gateway#commands-and-events-gateway-events">
20   *     https://discordapp.com/developers/docs/topics/gateway#commands-and-events-gateway-events</a>
21   */
22  public enum EventType {
23    GUILD_CREATE(GuildCreateData.class, ImmutableGuildCreate::of),
24    MESSAGE_CREATE(Message.class, ImmutableMessageCreate::of),
25    READY(ReadyData.class, ImmutableReady::of),
26    TYPING_START(TypingStartData.class, ImmutableTypingStart::of);
27  
28    private final EventFactory<?> factory;
29  
30    private EventType(EventFactory<?> factory) {
31      this.factory = factory;
32    }
33  
34    private <T> EventType(Class<T> dataClass, Function<T, Event<T>> factory) {
35      this(EventFactoryTuple.of(dataClass, factory));
36    }
37  
38    public EventFactory<?> getFactory() {
39      return factory;
40    }
41  }
