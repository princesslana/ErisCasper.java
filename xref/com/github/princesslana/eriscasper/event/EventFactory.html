7   @Value.Immutable
8   @Tuple
9   public abstract class EventFactory<D> {
10  
11    public abstract Class<D> getDataClass();
12  
13    protected abstract Function<D, Event<D>> getConstructor();
14  
15    /**
16     * Takes a data payload and constructs a given event from it.
17     *
18     * <p>It would be nice to have the types be more specific, but they are deliberately left generic
19     * due to difficulties in making the types work where this method is used ({@link
20     * com.github.princesslana.eriscasper.gateway.Payloads#toEvent(Event)})
21     *
22     * @param data data for the event to construct
23     * @return an event with the given data as the payload
24     * @throw ClassCastException if the data is not an instance of that given in {@link
25     *     #getDataClass()}
26     */
27    public Event<?> apply(Object data) {
28      return getConstructor().apply(getDataClass().cast(data));
29    }
30  }
