57    @Override
58    public RestAction<Application.Bot> createBot() {
59      if (this.hasBot()) return new RestAction.EmptyRestAction<>(getJDA(), this.bot);
60  
61      return new RestAction<Application.Bot>(
62          this.api, Route.Applications.CREATE_BOT.compile(getId())) {
63        @Override
64        protected void handleResponse(
65            final Response response, final Request<Application.Bot> request) {
66          if (response.isOk())
67            request.onSuccess(ApplicationImpl.this.bot = new BotImpl(response.getObject()));
68          else request.onFailure(response);
69        }
70      };
71    }

77        protected void handleResponse(final Response response, final Request<Void> request) {
78          if (response.isOk()) request.onSuccess(null);
79          else request.onFailure(response);

187     Route.CompiledRoute route = Route.Applications.RESET_BOT_TOKEN.compile(getId());
188     return new RestAction<Application>(this.api, route) {

203   public ApplicationImpl updateFromJson(final JSONObject object) {
204     if (object.has("bot")) {
205       final JSONObject botObject = object.getJSONObject("bot");
206 
207       if (this.bot == null) this.bot = new BotImpl(botObject);
208       else this.bot.updateFromJson(botObject);
209     } else {
210       this.bot = null;
211     }
212 
213     this.isBotPublic = object.getBoolean("bot_public");
214     this.doesBotRequireCodeGrant = object.getBoolean("bot_require_code_grant");
215     this.description = object.getString("description");
216     this.flags = object.getInt("flags");
217     this.iconId = object.has("icon") ? object.getString("icon") : null;
218     this.id = object.getLong("id");
219     this.name = object.getString("name");
220 
221     final JSONArray redirectUriArray = object.getJSONArray("redirect_uris");
222     if (this.redirectUris == null) this.redirectUris = new ArrayList<>(redirectUriArray.length());
223     else this.redirectUris.clear();
224 
225     for (int i = 0; i < redirectUriArray.length(); i++)
226       this.redirectUris.add(redirectUriArray.getString(i));
227 
228     this.rpcApplicationState = object.getInt("rpc_application_state");
229 
230     this.secret = object.getString("secret");
231 
232     return this;
233   }

288     @Override
289     public String getInviteUrl(final String guildId, final Collection<Permission> permissions) {
290       StringBuilder builder =
291           new StringBuilder("https://discordapp.com/oauth2/authorize?client_id=");
292       builder.append(this.getId());
293       builder.append("&scope=bot");
294       if (permissions != null && !permissions.isEmpty()) {
295         builder.append("&permissions=");
296         builder.append(Permission.getRaw(permissions));
297       }
298       if (guildId != null) {
299         builder.append("&guild_id=");
300         builder.append(guildId);
301       }
302       return builder.toString();
303     }
304 
305     @Override
306     public String getInviteUrl(final String guildId, final Permission... permissions) {
307       return this.getInviteUrl(guildId, permissions == null ? null : Arrays.asList(permissions));
308     }

331       Route.CompiledRoute route = Route.Applications.RESET_BOT_TOKEN.compile(getId());
332       return new RestAction<Bot>(getJDA(), route) {
333         @Override
334         protected void handleResponse(final Response response, final Request<Bot> request) {

336           else request.onFailure(response);
337         }
338       };
339     }

343       return "Application.Bot(" + this.id + ")";
344     }
345 
346     public BotImpl updateFromJson(final JSONObject object) {
347       this.name = object.getString("username");
348       this.discriminator = object.getString("discriminator");
349       this.token = object.getString("token");
350       this.id = object.getLong("id");
351       this.avatarId = object.has("avatar") ? object.getString("avatar") : null;
352 
353       return this;
354     }
