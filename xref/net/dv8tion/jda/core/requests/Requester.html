44    public static final String DISCORD_API_PREFIX =
45        String.format("https://discordapp.com/api/v%d/", JDAInfo.DISCORD_REST_VERSION);
46    public static final String USER_AGENT =
47        "DiscordBot (" + JDAInfo.GITHUB + ", " + JDAInfo.VERSION + ")";

59    public Requester(JDA api) {
60      this.api = (JDAImpl) api;
61      rateLimiter = new BotRateLimiter(this, 5);
62  
63      this.httpClient = this.api.getHttpClientBuilder().build();
64    }

66    public JDAImpl getJDA() {
67      return api;
68    }

70    public <T> void request(Request<T> apiRequest) {
71      if (rateLimiter.isShutdown)
72        throw new IllegalStateException(
73            "The Requester has been shutdown! No new requests can be requested!");
74  
75      if (apiRequest.shouldQueue()) rateLimiter.queueRequest(apiRequest);
76      else execute(apiRequest, false, true);
77    }
78  
79    public Long execute(Request<?> apiRequest) {
80      return execute(apiRequest, false, false);
81    }
82  
83    private Long execute(Request<?> apiRequest, boolean retried, boolean handleOnRatelimit) {
84      Route.CompiledRoute route = apiRequest.getRoute();
85      Long retryAfter = rateLimiter.getRateLimit(route);
86      if (retryAfter != null) {
87        if (handleOnRatelimit)
88          apiRequest.handleResponse(new Response(retryAfter, Collections.emptySet()));
89        return retryAfter;
90      }
91  
92      okhttp3.Request.Builder builder = new okhttp3.Request.Builder();
93  
94      String url = DISCORD_API_PREFIX + route.getCompiledRoute();
95      builder.url(url);
96  
97      String method = apiRequest.getRoute().getMethod().toString();
98      RequestBody body = apiRequest.getBody();
99  
100     if (body == null && HttpMethod.requiresRequestBody(method)) body = EMPTY_BODY;
101 
102     builder.method(method, body).header("user-agent", USER_AGENT).header("accept-encoding", "gzip");
103 
104     // adding token to all requests to the discord api or cdn pages
105     // we can check for startsWith(DISCORD_API_PREFIX) because the cdn endpoints don't need any kind
106     // of authorization
107     if (url.startsWith(DISCORD_API_PREFIX) && api.getToken() != null)
108       builder.header("authorization", api.getToken());
109 
110     // Apply custom headers like X-Audit-Log-Reason
111     // If customHeaders is null this does nothing
112     if (apiRequest.getHeaders() != null) {
113       for (Entry<String, String> header : apiRequest.getHeaders().entrySet())
114         builder.addHeader(header.getKey(), header.getValue());
115     }
116 
117     okhttp3.Request request = builder.build();
118 
119     Set<String> rays = new LinkedHashSet<>();
120     okhttp3.Response[] responses = new okhttp3.Response[4];
121     // we have an array of all responses to later close them all at once
122     // the response below this comment is used as the first successful response from the server
123     okhttp3.Response firstSuccess = null;
124     try {
125       int attempt = 0;
126       do {
127         // If the request has been canceled via the Future, don't execute.
128         // if (apiRequest.isCanceled())
129         //    return null;
130         Call call = httpClient.newCall(request);
131         firstSuccess = call.execute();
132         responses[attempt] = firstSuccess;
133         String cfRay = firstSuccess.header("CF-RAY");
134         if (cfRay != null) rays.add(cfRay);
135 
136         if (firstSuccess.code() < 500) break; // break loop, got a successful response!
137 
138         attempt++;
139         LOG.debug(
140             "Requesting {} -> {} returned status {}... retrying (attempt {})",
141             apiRequest.getRoute().getMethod(),
142             url,
143             firstSuccess.code(),
144             attempt);
145         try {
146           Thread.sleep(50 * attempt);
147         } catch (InterruptedException ignored) {
148         }
149       } while (attempt < 3 && firstSuccess.code() >= 500);
150 
151       if (firstSuccess.code() >= 500) {
152         // Epic failure from other end. Attempted 4 times.
153         return null;
154       }
155 
156       retryAfter = rateLimiter.handleResponse(route, firstSuccess);
157       if (!rays.isEmpty()) LOG.debug("Received response with following cf-rays: {}", rays);
158 
159       if (retryAfter == null) apiRequest.handleResponse(new Response(firstSuccess, -1, rays));
160       else if (handleOnRatelimit)
161         apiRequest.handleResponse(new Response(firstSuccess, retryAfter, rays));
162 
163       return retryAfter;
164     } catch (SocketTimeoutException e) {
165       if (retryOnTimeout && !retried) return execute(apiRequest, true, handleOnRatelimit);
166       LOG.error("Requester timed out while executing a request", e);
167       apiRequest.handleResponse(new Response(firstSuccess, e, rays));
168       return null;
169     } catch (Exception e) {
170       LOG.error(
171           "There was an exception while executing a REST request",
172           e); // This originally only printed on DEBUG in 2.x
173       apiRequest.handleResponse(new Response(firstSuccess, e, rays));
174       return null;
175     } finally {
176       for (okhttp3.Response r : responses) {
177         if (r == null) break;
178         r.close();
179       }
180     }
181   }

199   /**
200    * Retrieves an {@link java.io.InputStream InputStream} for the provided {@link okhttp3.Response
201    * Response}. <br>
202    * When the header for {@code content-encoding} is set with {@code gzip} this will wrap the body
203    * in a {@link java.util.zip.GZIPInputStream GZIPInputStream} which decodes the data.
204    *
205    * <p>This is used to make usage of encoded responses more user-friendly in various parts of JDA.
206    *
207    * @param response The not-null Response object
208    * @throws IOException If a GZIP format error has occurred or the compression method used is
209    *     unsupported
210    * @return InputStream representing the body of this response
211    */
212   public static InputStream getBody(okhttp3.Response response) throws IOException {
213     String encoding = response.header("content-encoding", "");
214     if (encoding.equals("gzip")) return new GZIPInputStream(response.body().byteStream());
215     return response.body().byteStream();
216   }
