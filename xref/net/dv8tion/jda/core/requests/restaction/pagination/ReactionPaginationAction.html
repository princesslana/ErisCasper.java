17  package net.dv8tion.jda.core.requests.restaction.pagination;
18  
19  import java.util.LinkedList;
20  import java.util.List;
21  import net.dv8tion.jda.core.entities.EntityBuilder;
22  import net.dv8tion.jda.core.entities.MessageReaction;
23  import net.dv8tion.jda.core.entities.User;
24  import net.dv8tion.jda.core.requests.Request;
25  import net.dv8tion.jda.core.requests.Response;
26  import net.dv8tion.jda.core.requests.Route;
27  import net.dv8tion.jda.core.utils.MiscUtil;
28  import org.json.JSONArray;
29  import org.json.JSONException;
30  
31  /**
32   * {@link net.dv8tion.jda.core.requests.restaction.pagination.PaginationAction PaginationAction}
33   * that paginates the endpoint {@link
34   * net.dv8tion.jda.core.requests.Route.Messages#GET_REACTION_USERS
35   * Route.Messages.GET_REACTION_USERS}.
36   *
37   * <p><b>Must provide not-null {@link net.dv8tion.jda.core.entities.MessageReaction MessageReaction}
38   * to compile a valid pagination route.</b>
39   *
40   * <h2>Limits:</h2>
41   *
42   * Minimum - 1 <br>
43   * Maximum - 100
44   *
45   * <h1>Example</h1>
46   *
47   * <pre><code>
48   * ReactionPaginationAction users = reaction.getUsers();
49   *
50   * {@literal Optional<User>} optUser = users.stream().skip(ThreadLocalRandom.current().nextInt(reaction.getCount())).findFirst();
51   * optUser.ifPresent( (user) {@literal ->} user.openPrivateChannel().queue(
52   *         (channel) {@literal ->} channel.sendMessage("I see you reacted to my message :eyes:").queue()
53   * ));
54   * </code></pre>
55   *
56   * @since 3.1
57   * @author Florian Spieß
58   */
59  public class ReactionPaginationAction extends PaginationAction<User, ReactionPaginationAction> {
60  
61    protected final MessageReaction reaction;
62  
63    /**
64     * Creates a new PaginationAction instance
65     *
66     * @param reaction The target {@link net.dv8tion.jda.core.entities.MessageReaction
67     *     MessageReaction}
68     */
69    public ReactionPaginationAction(MessageReaction reaction) {
70      super(
71          reaction.getJDA(),
72          Route.Messages.GET_REACTION_USERS.compile(
73              reaction.getChannel().getId(), reaction.getMessageId(), getCode(reaction)),
74          1,
75          100,
76          100);
77      this.reaction = reaction;
78    }
79  
80    protected static String getCode(MessageReaction reaction) {
81      MessageReaction.ReactionEmote emote = reaction.getReactionEmote();
82  
83      return emote.isEmote()
84          ? emote.getName() + ":" + emote.getId()
85          : MiscUtil.encodeUTF8(emote.getName());
86    }
87  
88    /**
89     * The current target {@link net.dv8tion.jda.core.entities.MessageReaction MessageReaction}
90     *
91     * @return The current MessageReaction
92     */
93    public MessageReaction getReaction() {
94      return reaction;
95    }
96  
97    @Override
98    protected Route.CompiledRoute finalizeRoute() {
99      Route.CompiledRoute route = super.finalizeRoute();
100 
101     String after = null;
102     String limit = String.valueOf(getLimit());
103     User last = this.last;
104     if (last != null) after = last.getId();
105 
106     route = route.withQueryParams("limit", limit);
107 
108     if (after != null) route = route.withQueryParams("after", after);
109 
110     return route;
111   }
112 
113   @Override
114   protected void handleResponse(Response response, Request<List<User>> request) {
115     if (!response.isOk()) {
116       request.onFailure(response);
117       return;
118     }
119 
120     final EntityBuilder builder = api.getEntityBuilder();
121     ;
122     final JSONArray array = response.getArray();
123     final List<User> users = new LinkedList<>();
124     for (int i = 0; i < array.length(); i++) {
125       try {
126         final User user = builder.createFakeUser(array.getJSONObject(i), false);
127         users.add(user);
128         if (useCache) cached.add(user);
129         last = user;
130       } catch (JSONException | NullPointerException e) {
131         LOG.warn("Encountered exception in ReactionPagination", e);
132       }
133     }
134 
135     request.onSuccess(users);
136   }
137 }
