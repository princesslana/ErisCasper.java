111   public WebSocketClient(JDAImpl api) {
112     this.api = api;
113     this.shardInfo = api.getShardInfo();
114     this.shouldReconnect = api.isAutoReconnect();
115     this.connectNode = new StartingNode();
116     try {
117       api.getSessionController().appendSession(connectNode);
118     } catch (RuntimeException | Error e) {
119       LOG.error("Failed to append new session to session controller queue. Shutting down!", e);
120       this.api.setStatus(JDA.Status.SHUTDOWN);
121       this.api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1006));
122       if (e instanceof RuntimeException) throw (RuntimeException) e;
123       else throw (Error) e;
124     }
125   }

162   public void ready() {
163     if (initiating) {
164       initiating = false;
165       processingReady = false;
166       if (firstInit) {
167         firstInit = false;
168         if (api.getGuilds().size() >= 2000) // Show large warning when connected to >=2000 guilds
169         {
170           JDAImpl.LOG.warn(" __      __ _    ___  _  _  ___  _  _   ___  _ ");
171           JDAImpl.LOG.warn(" \\ \\    / //_\\  | _ \\| \\| ||_ _|| \\| | / __|| |");
172           JDAImpl.LOG.warn("  \\ \\/\\/ // _ \\ |   /| .` | | | | .` || (_ ||_|");
173           JDAImpl.LOG.warn("   \\_/\\_//_/ \\_\\|_|_\\|_|\\_||___||_|\\_| \\___|(_)");
174           JDAImpl.LOG.warn("You're running a session with over 2000 connected");
175           JDAImpl.LOG.warn("guilds. You should shard the connection in order");
176           JDAImpl.LOG.warn("to split the load or things like resuming");
177           JDAImpl.LOG.warn("connection might not work as expected.");
178           JDAImpl.LOG.warn("For more info see https://git.io/vrFWP");
179         }
180         JDAImpl.LOG.info("Finished Loading!");
181         api.getEventManager().handle(new ReadyEvent(api, api.getResponseTotal()));
182       } else {
183         updateAudioManagerReferences();
184         JDAImpl.LOG.info("Finished (Re)Loading!");
185         api.getEventManager().handle(new ReconnectedEvent(api, api.getResponseTotal()));
186       }
187     } else {
188       JDAImpl.LOG.info("Successfully resumed Session!");
189       api.getEventManager().handle(new ResumedEvent(api, api.getResponseTotal()));
190     }
191     api.setStatus(JDA.Status.CONNECTED);
192     LOG.debug("Resending {} cached events...", cachedEvents.size());
193     handle(cachedEvents);
194     LOG.debug("Sending of cached events finished.");
195     cachedEvents.clear();
196   }

214   private boolean send(String message, boolean skipQueue) {
215     if (!connected) return false;
216 
217     long now = System.currentTimeMillis();
218 
219     if (this.ratelimitResetTime <= now) {
220       this.messagesSent = 0;
221       this.ratelimitResetTime = now + 60000; // 60 seconds
222       this.printedRateLimitMessage = false;
223     }
224 
225     // Allows 115 messages to be sent before limiting.
226     if (this.messagesSent <= 115
227         || (skipQueue
228             && this.messagesSent
229                 <= 119)) // technically we could go to 120, but we aren't going to chance it
230     {
231       LOG.trace("<- {}", message);
232       socket.sendText(message);
233       this.messagesSent++;
234       return true;
235     } else {
236       if (!printedRateLimitMessage) {
237         LOG.warn(
238             "Hit the WebSocket RateLimit! If you see this message a lot then you might need to talk to DV8FromTheWorld.");
239         printedRateLimitMessage = true;
240       }
241       return false;
242     }
243   }

245   private void setupSendingThread() {
246     ratelimitThread =
247         new Thread(
248             () -> {
249               boolean needRatelimit;
250               boolean attemptedToSend;
251               while (!Thread.currentThread().isInterrupted()) {
252                 try {
253                   // Make sure that we don't send any packets before sending auth info.
254                   if (!sentAuthInfo) {
255                     Thread.sleep(500);
256                     continue;
257                   }
258                   attemptedToSend = false;
259                   needRatelimit = false;
260                   audioQueueLock.lockInterruptibly();
261 
262                   ConnectionRequest audioRequest = getNextAudioConnectRequest();
263                   String chunkOrSyncRequest = chunkSyncQueue.peekFirst();
264 
265                   // if lock isn't needed we already unlock here
266                   if (audioRequest == null || chunkOrSyncRequest != null) maybeUnlock();
267                   if (chunkOrSyncRequest != null) {
268                     needRatelimit = !send(chunkOrSyncRequest, false);
269                     if (!needRatelimit) chunkSyncQueue.removeFirst();
270 
271                     attemptedToSend = true;
272                   } else if (audioRequest != null) {
273                     VoiceChannel channel = audioRequest.getChannel();
274                     Guild guild = api.getGuildById(audioRequest.getGuildIdLong());
275                     if (guild == null) {
276                       // race condition on guild delete, avoid NPE on DISCONNECT requests
277                       queuedAudioConnections.remove(audioRequest.getGuildIdLong());
278 
279                       maybeUnlock();
280                       continue;
281                     }
282                     ConnectionStage stage = audioRequest.getStage();
283                     AudioManager audioManager = guild.getAudioManager();
284                     JSONObject packet;
285                     switch (stage) {
286                       case RECONNECT:
287                       case DISCONNECT:
288                         packet = newVoiceClose(audioRequest.getGuildIdLong());
289                         break;
290                       default:
291                       case CONNECT:
292                         packet = newVoiceOpen(audioManager, channel);
293                     }
294                     needRatelimit = !send(packet.toString(), false);
295                     if (!needRatelimit) {
296                       // If we didn't get RateLimited, Next request attempt will be 2 seconds from
297                       // now
298                       // we remove it in VoiceStateUpdateHandler once we hear that it has updated
299                       // our status
300                       // in 2 seconds we will attempt again in case we did not receive an update
301                       audioRequest.setNextAttemptEpoch(System.currentTimeMillis() + 2000);
302                       // If we are already in the correct state according to voice state
303                       // we will not receive a VOICE_STATE_UPDATE that would remove it
304                       // thus we update it here
305                       final GuildVoiceState voiceState = guild.getSelfMember().getVoiceState();
306                       updateAudioConnection0(guild.getIdLong(), voiceState.getChannel());
307                     }
308                     maybeUnlock();
309                     attemptedToSend = true;
310                   } else {
311                     String message = ratelimitQueue.peekFirst();
312                     if (message != null) {
313                       needRatelimit = !send(message, false);
314                       if (!needRatelimit) ratelimitQueue.removeFirst();
315                       attemptedToSend = true;
316                     }
317                   }
318 
319                   if (needRatelimit || !attemptedToSend) Thread.sleep(1000);
320                 } catch (InterruptedException ignored) {
321                   LOG.debug(
322                       "Main WS send thread interrupted. Most likely JDA is disconnecting the websocket.");
323                   break;
324                 } finally {
325                   // on any exception that might cause this lock to not release
326                   maybeUnlock();
327                 }
328               }
329             });
330     ratelimitThread.setUncaughtExceptionHandler(
331         (thread, throwable) -> {
332           handleCallbackError(socket, throwable);
333           setupSendingThread();
334         });
335     ratelimitThread.setName(api.getIdentifierString() + " MainWS-Sending Thread");
336     ratelimitThread.start();
337   }

386   protected synchronized void connect() {
387     if (api.getStatus() != JDA.Status.ATTEMPTING_TO_RECONNECT)
388       api.setStatus(JDA.Status.CONNECTING_TO_WEBSOCKET);
389     if (shutdown) throw new RejectedExecutionException("JDA is shutdown!");
390     initiating = true;
391 
392     String url =
393         api.getGatewayUrl() + "?encoding=json&compress=zlib-stream&v=" + DISCORD_GATEWAY_VERSION;
394 
395     try {
396       socket =
397           api.getWebSocketFactory()
398               .createSocket(url)
399               .addHeader("Accept-Encoding", "gzip")
400               .addListener(this);
401       socket.connect();
402     } catch (IOException | WebSocketException e) {
403       // Completely fail here. We couldn't make the connection.
404       throw new IllegalStateException(e);
405     }
406   }

418   @Override
419   public void onConnected(WebSocket websocket, Map<String, List<String>> headers) {
420     // writing thread
421     api.setStatus(JDA.Status.IDENTIFYING_SESSION);
422     LOG.info("Connected to WebSocket");
423     if (headers.containsKey("cf-ray")) {
424       List<String> values = headers.get("cf-ray");
425       if (!values.isEmpty()) {
426         String ray = values.get(0);
427         cfRays.add(ray);
428         LOG.debug("Received new CF-RAY: {}", ray);
429       }
430     }
431     connected = true;
432     reconnectTimeoutS = 2;
433     messagesSent = 0;
434     ratelimitResetTime = System.currentTimeMillis() + 60000;
435     if (sessionId == null) sendIdentify();
436     else sendResume();
437   }

439   @Override
440   public void onDisconnected(
441       WebSocket websocket,
442       WebSocketFrame serverCloseFrame,
443       WebSocketFrame clientCloseFrame,
444       boolean closedByServer) {
445     sentAuthInfo = false;
446     connected = false;
447     api.setStatus(JDA.Status.DISCONNECTED);
448 
449     CloseCode closeCode = null;
450     int rawCloseCode = 1000;
451     // When we get 1000 from remote close we will try to resume
452     // as apparently discord doesn't understand what "graceful disconnect" means
453     boolean isInvalidate = false;
454 
455     if (keepAliveThread != null) {
456       keepAliveThread.interrupt();
457       keepAliveThread = null;
458     }
459     if (serverCloseFrame != null) {
460       rawCloseCode = serverCloseFrame.getCloseCode();
461       closeCode = CloseCode.from(rawCloseCode);
462       if (closeCode == CloseCode.RATE_LIMITED)
463         LOG.error(
464             "WebSocket connection closed due to ratelimit! Sent more than 120 websocket messages in under 60 seconds!");
465       else if (closeCode != null) LOG.debug("WebSocket connection closed with code {}", closeCode);
466       else
467         LOG.warn(
468             "WebSocket connection closed with unknown meaning for close-code {}", rawCloseCode);
469     }
470     if (clientCloseFrame != null
471         && clientCloseFrame.getCloseCode() == 1000
472         && Objects.equals(clientCloseFrame.getCloseReason(), INVALIDATE_REASON)) {
473       // When we close with 1000 we properly dropped our session due to invalidation
474       // in that case we can be sure that resume will not work and instead we invalidate and
475       // reconnect here
476       isInvalidate = true;
477     }
478 
479     // null is considered -reconnectable- as we do not know the close-code meaning
480     boolean closeCodeIsReconnect = closeCode == null || closeCode.isReconnect();
481     if (!shouldReconnect || !closeCodeIsReconnect) // we should not reconnect
482     {
483       if (ratelimitThread != null) ratelimitThread.interrupt();
484 
485       if (!closeCodeIsReconnect) {
486         // it is possible that a token can be invalidated due to too many reconnect attempts
487         // or that a bot reached a new shard minimum and cannot connect with the current settings
488         // if that is the case we have to drop our connection and inform the user with a fatal error
489         // message
490         LOG.error(
491             "WebSocket connection was closed and cannot be recovered due to identification issues\n{}",
492             closeCode);
493       }
494 
495       api.setStatus(JDA.Status.SHUTDOWN);
496       api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), rawCloseCode));
497     } else {
498       // reset our zlib decompression tools
499       zlibContext = new Inflater();
500       readBuffer = null;
501       if (isInvalidate) invalidate(); // 1000 means our session is dropped so we cannot resume
502       api.getEventManager()
503           .handle(
504               new DisconnectEvent(
505                   api, serverCloseFrame, clientCloseFrame, closedByServer, OffsetDateTime.now()));
506       try {
507         if (sessionId == null) queueReconnect();
508         else // if resume is possible
509         reconnect();
510       } catch (InterruptedException e) {
511         LOG.error("Failed to resume due to interrupted thread", e);
512         invalidate();
513         queueReconnect();
514       }
515     }
516   }

518   protected void queueReconnect() {
519     if (!handleIdentifyRateLimit)
520       LOG.warn(
521           "Got disconnected from WebSocket (Internet?!)... Appending session to reconnect queue");
522     try {
523       this.api.setStatus(JDA.Status.RECONNECT_QUEUED);
524       this.connectNode = new ReconnectNode();
525       this.api.getSessionController().appendSession(connectNode);
526     } catch (IllegalStateException ex) {
527       LOG.error("Reconnect queue rejected session. Shutting down...");
528       this.api.setStatus(JDA.Status.SHUTDOWN);
529       this.api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1006));
530     }
531   }

537   /**
538    * This method is used to start the reconnect of the JDA instance. It is public for access from
539    * SessionReconnectQueue extensions.
540    *
541    * @param callFromQueue whether this was in SessionReconnectQueue and got polled
542    * @param shouldHandleIdentify whether SessionReconnectQueue already handled an IDENTIFY rate
543    *     limit for this session
544    */
545   public void reconnect(boolean callFromQueue, boolean shouldHandleIdentify)
546       throws InterruptedException {
547     if (shutdown) {
548       api.setStatus(JDA.Status.SHUTDOWN);
549       api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1000));
550       return;
551     }
552     if (!handleIdentifyRateLimit) {
553       if (callFromQueue)
554         LOG.warn(
555             "Queue is attempting to reconnect a shard...{}",
556             shardInfo != null ? " Shard: " + shardInfo.getShardString() : "");
557       else LOG.warn("Got disconnected from WebSocket (Internet?!)...");
558       LOG.warn("Attempting to reconnect in {}s", reconnectTimeoutS);
559     }
560     while (shouldReconnect) {
561       api.setStatus(JDA.Status.WAITING_TO_RECONNECT);
562       int delay = IDENTIFY_DELAY;
563       if (handleIdentifyRateLimit && shouldHandleIdentify)
564         LOG.error(
565             "Encountered IDENTIFY (OP {}) Rate Limit! Waiting {} seconds before trying again!",
566             WebSocketCode.IDENTIFY,
567             IDENTIFY_DELAY);
568       else delay = reconnectTimeoutS;
569       Thread.sleep(delay * 1000);
570       handleIdentifyRateLimit = false;
571       api.setStatus(JDA.Status.ATTEMPTING_TO_RECONNECT);
572       LOG.warn("Attempting to reconnect!");
573       try {
574         connect();
575         break;
576       } catch (RejectedExecutionException ex) {
577         // JDA has already been shutdown so we can stop here
578         api.setStatus(JDA.Status.SHUTDOWN);
579         api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1000));
580         return;
581       } catch (RuntimeException ex) {
582         reconnectTimeoutS = Math.min(reconnectTimeoutS << 1, api.getMaxReconnectDelay());
583         LOG.warn("Reconnect failed! Next attempt in {}s", reconnectTimeoutS);
584       }
585     }
586   }

588   @Override
589   public void onTextMessage(WebSocket websocket, String message) {
590     // reading thread
591     JSONObject content = new JSONObject(message);
592     try {
593       int opCode = content.getInt("op");
594 
595       if (!content.isNull("s")) {
596         api.setResponseTotal(content.getInt("s"));
597       }
598 
599       switch (opCode) {
600         case WebSocketCode.DISPATCH:
601           handleEvent(content);
602           break;
603         case WebSocketCode.HEARTBEAT:
604           LOG.debug("Got Keep-Alive request (OP 1). Sending response...");
605           sendKeepAlive();
606           break;
607         case WebSocketCode.RECONNECT:
608           LOG.debug("Got Reconnect request (OP 7). Closing connection now...");
609           close(4000, "OP 7: RECONNECT");
610           break;
611         case WebSocketCode.INVALIDATE_SESSION:
612           LOG.debug("Got Invalidate request (OP 9). Invalidating...");
613           sentAuthInfo = false;
614           final boolean isResume = content.getBoolean("d");
615           // When d: true we can wait a bit and then try to resume again
616           // sending 4000 to not drop session
617           int closeCode = isResume ? 4000 : 1000;
618           if (isResume)
619             LOG.debug("Session can be recovered... Closing and sending new RESUME request");
620           else invalidate();
621 
622           close(closeCode, INVALIDATE_REASON);
623           break;
624         case WebSocketCode.HELLO:
625           LOG.debug("Got HELLO packet (OP 10). Initializing keep-alive.");
626           final JSONObject data = content.getJSONObject("d");
627           setupKeepAlive(data.getLong("heartbeat_interval"));
628           if (!data.isNull("_trace"))
629             updateTraces(data.getJSONArray("_trace"), "HELLO", WebSocketCode.HELLO);
630           break;
631         case WebSocketCode.HEARTBEAT_ACK:
632           LOG.trace("Got Heartbeat Ack (OP 11).");
633           api.setPing(System.currentTimeMillis() - heartbeatStartTime);
634           break;
635         default:
636           LOG.debug("Got unknown op-code: {} with content: {}", opCode, message);
637       }
638     } catch (Exception ex) {
639       LOG.error("Encountered exception on lifecycle level\nJSON: {}", content, ex);
640       api.getEventManager().handle(new ExceptionEvent(api, ex, true));
641     }
642   }

644   protected void setupKeepAlive(long timeout) {
645     keepAliveThread =
646         new Thread(
647             () -> {
648               while (connected) {
649                 try {
650                   sendKeepAlive();
651 
652                   // Sleep for heartbeat interval
653                   Thread.sleep(timeout);
654                 } catch (InterruptedException ex) {
655                   // connection got cut... terminating keepAliveThread
656                   break;
657                 }
658               }
659             });
660     keepAliveThread.setUncaughtExceptionHandler(
661         (thread, throwable) -> {
662           handleCallbackError(socket, throwable);
663           setupKeepAlive(timeout);
664         });
665     keepAliveThread.setName(api.getIdentifierString() + " MainWS-KeepAlive Thread");
666     keepAliveThread.setPriority(Thread.MAX_PRIORITY);
667     keepAliveThread.setDaemon(true);
668     keepAliveThread.start();
669   }

783           newMng.setReceivingHandler(mng.getReceiveHandler());
784           newMng.setConnectionListener(listener);
785           newMng.setAutoReconnect(mng.isAutoReconnect());
786 
787           if (mng.isConnected() || mng.isAttemptingToConnect()) {
788             final long channelId =
789                 mng.isConnected()
790                     ? mng.getConnectedChannel().getIdLong()
791                     : mng.getQueuedAudioConnection().getIdLong();
792 
793             final VoiceChannel channel = api.getVoiceChannelById(channelId);
794             if (channel != null) {

796               // closing old connection in order to reconnect later
797               newMng.setQueuedAudioConnection(channel);
798             } else {
799               // The voice channel is not cached. It was probably deleted.
800               queuedAudioConnections.remove(guildId);
801               if (listener != null)
802                 listener.onStatusChange(ConnectionStatus.DISCONNECTED_CHANNEL_DELETED);
803             }
804           }
805           it.setValue(newMng);
806         }
807       }
808     }

832   protected void handleEvent(JSONObject raw) {
833     String type = raw.getString("t");
834     long responseTotal = api.getResponseTotal();
835 
836     if (type.equals("GUILD_MEMBER_ADD"))
837       ((GuildMembersChunkHandler) getHandler("GUILD_MEMBERS_CHUNK"))
838           .modifyExpectedGuildMember(raw.getJSONObject("d").getLong("guild_id"), 1);
839     if (type.equals("GUILD_MEMBER_REMOVE"))
840       ((GuildMembersChunkHandler) getHandler("GUILD_MEMBERS_CHUNK"))
841           .modifyExpectedGuildMember(raw.getJSONObject("d").getLong("guild_id"), -1);
842 
843     boolean isJSON = raw.opt("d") instanceof JSONObject;
844     // If initiating, only allows READY, RESUMED, GUILD_MEMBERS_CHUNK, GUILD_SYNC, and GUILD_CREATE
845     // through.
846     // If we are currently chunking, we don't allow GUILD_CREATE through anymore.
847     if (initiating
848         && !(type.equals("READY")
849             || type.equals("GUILD_MEMBERS_CHUNK")
850             || type.equals("RESUMED")
851             || type.equals("GUILD_SYNC")
852             || (!chunkingAndSyncing && type.equals("GUILD_CREATE")))) {
853       if (!isJSON) {
854         if (type.equals("PRESENCES_REPLACE")) {
855           List<JSONObject> converted =
856               convertPresencesReplace(responseTotal, raw.getJSONArray("d"));
857           LOG.debug("Caching PRESENCES_REPLACE event during init as PRESENCE_UPDATE dispatches!");
858           cachedEvents.addAll(converted);
859         } else {
860           LOG.debug("Received event with unhandled body type during init JSON: {}", raw);
861         }
862         return;
863       }
864       // If we are currently GuildStreaming, and we get a GUILD_DELETE informing us that a Guild is
865       // unavailable
866       // convert it to a GUILD_CREATE for handling.
867       JSONObject content = raw.getJSONObject("d");
868       if (!chunkingAndSyncing
869           && type.equals("GUILD_DELETE")
870           && content.has("unavailable")
871           && content.getBoolean("unavailable")) {
872         type = "GUILD_CREATE";
873         raw.put("t", "GUILD_CREATE")
874             .put(
875                 "jda-field",
876                 "This event was originally a GUILD_DELETE but was converted to GUILD_CREATE for WS init Guild streaming");
877       } else {
878         LOG.debug("Caching {} event during init!", type);
879         cachedEvents.add(raw);
880         return;
881       }
882     }
883 
884     if (!isJSON) {
885       // Needs special handling due to content of "d" being an array
886       if (type.equals("PRESENCES_REPLACE")) {
887         final JSONArray payload = raw.getJSONArray("d");
888         final List<JSONObject> converted = convertPresencesReplace(responseTotal, payload);
889         final PresenceUpdateHandler handler = getHandler("PRESENCE_UPDATE");
890         LOG.trace(String.format("%s -> %s", type, payload.toString()));
891         for (JSONObject o : converted) handler.handle(responseTotal, o);
892       } else {
893         LOG.debug("Received event with unhandled body type JSON: {}", raw);
894       }
895       return;
896     }
897 
898     JSONObject content = raw.getJSONObject("d");
899     LOG.trace("{} -> {}", type, content);
900 
901     try {
902       switch (type) {
903           // INIT types
904         case "READY":
905           api.setStatus(JDA.Status.LOADING_SUBSYSTEMS);
906           processingReady = true;
907           handleIdentifyRateLimit = false;
908           sessionId = content.getString("session_id");
909           if (!content.isNull("_trace"))
910             updateTraces(content.getJSONArray("_trace"), "READY", WebSocketCode.DISPATCH);
911           handlers.get("READY").handle(responseTotal, raw);
912           break;
913         case "RESUMED":
914           sentAuthInfo = true;
915           if (!processingReady) {
916             api.setStatus(JDA.Status.LOADING_SUBSYSTEMS);
917             initiating = false;
918             ready();
919           }
920           if (!content.isNull("_trace"))
921             updateTraces(content.getJSONArray("_trace"), "RESUMED", WebSocketCode.DISPATCH);
922           break;
923         default:
924           SocketHandler handler = handlers.get(type);
925           if (handler != null) handler.handle(responseTotal, raw);
926           else LOG.debug("Unrecognized event:\n{}", raw);
927       }
928     } catch (JSONException ex) {
929       LOG.warn(
930           "Got an unexpected Json-parse error. Please redirect following message to the devs:\n\t{}\n\t{} -> {}",
931           ex.getMessage(),
932           type,
933           content,
934           ex);
935     } catch (Exception ex) {
936       LOG.error(
937           "Got an unexpected error. Please redirect following message to the devs:\n\t{} -> {}",
938           type,
939           content,
940           ex);
941     }
942   }

944   protected boolean onBufferMessage(byte[] binary) throws IOException {
945     if (binary.length >= 4 && getInt(binary, binary.length - 4) == ZLIB_SUFFIX) {
946       extendBuffer(binary);
947       return true;
948     }
949 
950     if (readBuffer != null) extendBuffer(binary);
951     else allocateBuffer(binary);
952 
953     return false;
954   }

956   @Override
957   public void onBinaryMessage(WebSocket websocket, byte[] binary)
958       throws IOException, DataFormatException {
959     if (!onBufferMessage(binary)) return;
960     // Thanks to ShadowLordAlpha and Shredder121 for code and debugging.
961     // Get the compressed message and inflate it
962     final int size = readBuffer != null ? readBuffer.size() : binary.length;
963     ByteArrayOutputStream out = new ByteArrayOutputStream(size * 2);
964     try (InflaterOutputStream decompressor = new InflaterOutputStream(out, zlibContext)) {
965       if (readBuffer != null) readBuffer.writeTo(decompressor);
966       else decompressor.write(binary);
967       // send the inflated message to the TextMessage method
968       onTextMessage(websocket, out.toString("UTF-8"));
969     } catch (IOException e) {
970       throw (DataFormatException) new DataFormatException("Malformed").initCause(e);
971     } finally {
972       readBuffer = null;
973     }
974   }

1120   public ConnectionRequest updateAudioConnection0(long guildId, VoiceChannel connectedChannel) {
1121     // Called by VoiceStateUpdateHandler when we receive a response from discord
1122     // about our request to CONNECT or DISCONNECT.
1123     // "stage" should never be RECONNECT here thus we don't check for that case
1124     ConnectionRequest request = queuedAudioConnections.get(guildId);
1125 
1126     if (request == null) return null;
1127     ConnectionStage requestStage = request.getStage();
1128     if (connectedChannel == null) {
1129       // If we got an update that DISCONNECT happened
1130       // -> If it was on RECONNECT we now switch to CONNECT
1131       // -> If it was on DISCONNECT we can now remove it
1132       // -> Otherwise we ignore it
1133       switch (requestStage) {
1134         case DISCONNECT:
1135           return queuedAudioConnections.remove(guildId);
1136         case RECONNECT:
1137           request.setStage(ConnectionStage.CONNECT);
1138           request.setNextAttemptEpoch(System.currentTimeMillis());
1139         default:
1140           return null;
1141       }
1142     } else if (requestStage == ConnectionStage.CONNECT) {
1143       // If the removeRequest was related to a channel that isn't the currently queued
1144       // request, then don't remove it.
1145       if (request.getChannel().getIdLong() == connectedChannel.getIdLong())
1146         return queuedAudioConnections.remove(guildId);
1147     }
1148     // If the channel is not the one we are looking for!
1149     return null;
1150   }

1157   protected ConnectionRequest getNextAudioConnectRequest() {
1158     // Don't try to setup audio connections before JDA has finished loading.
1159     if (!isReady()) return null;
1160 
1161     long now = System.currentTimeMillis();
1162     TLongObjectIterator<ConnectionRequest> it = queuedAudioConnections.iterator();
1163     while (it.hasNext()) {
1164       it.advance();
1165       ConnectionRequest audioRequest = it.value();
1166       if (audioRequest.getNextAttemptEpoch() < now) {
1167         Guild guild = api.getGuildById(audioRequest.getGuildIdLong());
1168         if (guild == null) {
1169           it.remove();
1170           // if (listener != null)
1171           //    listener.onStatusChange(ConnectionStatus.DISCONNECTED_REMOVED_FROM_GUILD);
1172           // already handled by event handling
1173           continue;
1174         }
1175 
1176         ConnectionListener listener = guild.getAudioManager().getConnectionListener();
1177         if (audioRequest.getStage() != ConnectionStage.DISCONNECT) {
1178           VoiceChannel channel = guild.getVoiceChannelById(audioRequest.getChannel().getIdLong());
1179           if (channel == null) {
1180             it.remove();
1181             if (listener != null)
1182               listener.onStatusChange(ConnectionStatus.DISCONNECTED_CHANNEL_DELETED);
1183             continue;
1184           }
1185 
1186           if (!guild.getSelfMember().hasPermission(channel, Permission.VOICE_CONNECT)) {
1187             it.remove();
1188             if (listener != null)
1189               listener.onStatusChange(ConnectionStatus.DISCONNECTED_LOST_PERMISSION);
1190             continue;
1191           }
1192         }
1193 
1194         return audioRequest;
1195       }
1196     }
1197 
1198     return null;
1199   }
