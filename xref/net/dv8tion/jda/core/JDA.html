37    /** Represents the connection status of JDA and its Main WebSocket. */
38    enum Status {
39      /** JDA is currently setting up supporting systems like the AudioSystem. */
40      INITIALIZING(true),
41      /** JDA has finished setting up supporting systems and is ready to log in. */
42      INITIALIZED(true),
43      /** JDA is currently attempting to log in. */
44      LOGGING_IN(true),
45      /** JDA is currently attempting to connect it's websocket to Discord. */
46      CONNECTING_TO_WEBSOCKET(true),
47      /** JDA has successfully connected it's websocket to Discord and is sending authentication */
48      IDENTIFYING_SESSION(true),
49      /** JDA has sent authentication to discord and is awaiting confirmation */
50      AWAITING_LOGIN_CONFIRMATION(true),
51      /**
52       * JDA is populating internal objects. This process often takes the longest of all Statuses
53       * (besides CONNECTED)
54       */
55      LOADING_SUBSYSTEMS(true),
56      /**
57       * JDA has finished loading everything, is receiving information from Discord and is firing
58       * events.
59       */
60      CONNECTED(true),
61      /**
62       * JDA's main websocket has been disconnected. This <b>DOES NOT</b> mean JDA has shutdown
63       * permanently. This is an in-between status. Most likely ATTEMPTING_TO_RECONNECT or
64       * SHUTTING_DOWN/SHUTDOWN will soon follow.
65       */
66      DISCONNECTED,
67      /**
68       * JDA session has been added to {@link net.dv8tion.jda.core.utils.SessionController
69       * SessionController} and is awaiting to be dequeued for reconnecting.
70       */
71      RECONNECT_QUEUED,
72      /**
73       * When trying to reconnect to Discord JDA encountered an issue, most likely related to a lack
74       * of internet connection, and is waiting to try reconnecting again.
75       */
76      WAITING_TO_RECONNECT,
77      /**
78       * JDA has been disconnected from Discord and is currently trying to reestablish the connection.
79       */
80      ATTEMPTING_TO_RECONNECT,
81      /**
82       * JDA has received a shutdown request or has been disconnected from Discord and reconnect is
83       * disabled, thus, JDA is in the process of shutting down
84       */
85      SHUTTING_DOWN,
86      /**
87       * JDA has finished shutting down and this instance can no longer be used to communicate with
88       * the Discord servers.
89       */
90      SHUTDOWN,
91      /**
92       * While attempting to authenticate, Discord reported that the provided authentication
93       * information was invalid.
94       */
95      FAILED_TO_LOGIN;
96  
97      private final boolean isInit;
98  
99      Status(boolean isInit) {
100       this.isInit = isInit;
101     }
102 
103     Status() {
104       this.isInit = false;
105     }
106 
107     public boolean isInit() {
108       return isInit;
109     }
110   }

112   /** Represents the information used to create this shard. */
113   class ShardInfo {
114     int shardId;
115     int shardTotal;
116 
117     public ShardInfo(int shardId, int shardTotal) {
118       this.shardId = shardId;
119       this.shardTotal = shardTotal;
120     }
121 
122     /**
123      * Represents the id of the shard of the current instance. <br>
124      * This value will be between 0 and ({@link #getShardTotal()} - 1).
125      *
126      * @return The id of the currently logged in shard.
127      */
128     public int getShardId() {
129       return shardId;
130     }
131 
132     /**
133      * The total amount of shards based on the value provided during JDA instance creation using
134      * {@link JDABuilder#useSharding(int, int)}. <br>
135      * This <b>does not</b> query Discord to determine the total number of shards. <br>
136      * This <b>does not</b> represent the amount of logged in shards. <br>
137      * It strictly represents the integer value provided to discord representing the total amount of
138      * shards that the developer indicated that it was going to use when initially starting JDA.
139      *
140      * @return The total of shards based on the total provided by the developer during JDA
141      *     initialization.
142      */
143     public int getShardTotal() {
144       return shardTotal;
145     }
146 
147     /**
148      * Provides a shortcut method for easily printing shard info. <br>
149      * Format: "[# / #]" <br>
150      * Where the first # is shardId and the second # is shardTotal.
151      *
152      * @return A String representing the information used to build this shard.
153      */
154     public String getShardString() {
155       return "[" + shardId + " / " + shardTotal + "]";
156     }
157 
158     @Override
159     public String toString() {
160       return "Shard " + getShardString();
161     }
162 
163     @Override
164     public boolean equals(Object o) {
165       if (!(o instanceof ShardInfo)) return false;
166 
167       ShardInfo oInfo = (ShardInfo) o;
168       return shardId == oInfo.getShardId() && shardTotal == oInfo.getShardTotal();
169     }
170   }
