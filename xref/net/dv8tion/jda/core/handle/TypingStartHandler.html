17  package net.dv8tion.jda.core.handle;
18  
19  import java.time.Instant;
20  import java.time.OffsetDateTime;
21  import java.time.ZoneOffset;
22  import net.dv8tion.jda.client.entities.Group;
23  import net.dv8tion.jda.client.entities.impl.GroupImpl;
24  import net.dv8tion.jda.core.AccountType;
25  import net.dv8tion.jda.core.entities.MessageChannel;
26  import net.dv8tion.jda.core.entities.PrivateChannel;
27  import net.dv8tion.jda.core.entities.TextChannel;
28  import net.dv8tion.jda.core.entities.User;
29  import net.dv8tion.jda.core.entities.impl.JDAImpl;
30  import net.dv8tion.jda.core.events.user.UserTypingEvent;
31  import org.json.JSONObject;
32  
33  public class TypingStartHandler extends SocketHandler {
34  
35    public TypingStartHandler(JDAImpl api) {
36      super(api);
37    }
38  
39    @Override
40    protected Long handleInternally(JSONObject content) {
41      final long channelId = content.getLong("channel_id");
42      MessageChannel channel = api.getTextChannelMap().get(channelId);
43      if (channel == null) channel = api.getPrivateChannelMap().get(channelId);
44      if (channel == null) channel = api.getFakePrivateChannelMap().get(channelId);
45      if (channel == null && api.getAccountType() == AccountType.CLIENT)
46        channel = api.asClient().getGroupById(channelId);
47      if (channel == null)
48        return null; // We don't have the channel cached yet. We chose not to cache this event
49      // because that happen very often and could easily fill up the EventCache if
50      // we, for some reason, never get the channel. Especially in an active channel.
51  
52      if (channel instanceof TextChannel) {
53        final long guildId = ((TextChannel) channel).getGuild().getIdLong();
54        if (api.getGuildLock().isLocked(guildId)) return guildId;
55      }
56  
57      final long userId = content.getLong("user_id");
58      User user;
59      if (channel instanceof PrivateChannel) user = ((PrivateChannel) channel).getUser();
60      else if (channel instanceof Group) user = ((GroupImpl) channel).getUserMap().get(userId);
61      else user = api.getUserMap().get(userId);
62  
63      if (user == null)
64        return null; // Just like in the comment above, if for some reason we don't have the user for
65      // some reason
66      // then we will just throw the event away.
67  
68      OffsetDateTime timestamp =
69          Instant.ofEpochSecond(content.getInt("timestamp")).atOffset(ZoneOffset.UTC);
70      api.getEventManager()
71          .handle(new UserTypingEvent(api, responseNumber, user, channel, timestamp));
72      return null;
73    }
74  }
