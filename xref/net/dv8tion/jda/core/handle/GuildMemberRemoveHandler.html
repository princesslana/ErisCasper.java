17  package net.dv8tion.jda.core.handle;
18  
19  import net.dv8tion.jda.core.entities.VoiceChannel;
20  import net.dv8tion.jda.core.entities.impl.*;
21  import net.dv8tion.jda.core.events.guild.member.GuildMemberLeaveEvent;
22  import net.dv8tion.jda.core.events.guild.voice.GuildVoiceLeaveEvent;
23  import net.dv8tion.jda.core.requests.WebSocketClient;
24  import org.json.JSONObject;
25  
26  public class GuildMemberRemoveHandler extends SocketHandler {
27  
28    public GuildMemberRemoveHandler(JDAImpl api) {
29      super(api);
30    }
31  
32    @Override
33    protected Long handleInternally(JSONObject content) {
34      final long id = content.getLong("guild_id");
35      if (api.getGuildLock().isLocked(id)) return id;
36  
37      GuildImpl guild = (GuildImpl) api.getGuildMap().get(id);
38      if (guild == null) {
39        // We probably just left the guild and this event is trying to remove us from the guild,
40        // therefore ignore
41        return null;
42      }
43  
44      final long userId = content.getJSONObject("user").getLong("id");
45      if (userId == api.getSelfUser().getIdLong()) {
46        // We probably just left the guild and this event is trying to remove us from the guild,
47        // therefore ignore
48        return null;
49      }
50      MemberImpl member = (MemberImpl) guild.getMembersMap().remove(userId);
51  
52      if (member == null) {
53        WebSocketClient.LOG.debug(
54            "Received GUILD_MEMBER_REMOVE for a Member that does not exist in the specified Guild.");
55        return null;
56      }
57  
58      if (member
59          .getVoiceState()
60          .inVoiceChannel()) // If this user was in a VoiceChannel, fire VoiceLeaveEvent.
61      {
62        GuildVoiceStateImpl vState = (GuildVoiceStateImpl) member.getVoiceState();
63        VoiceChannel channel = vState.getChannel();
64        vState.setConnectedChannel(null);
65        ((VoiceChannelImpl) channel).getConnectedMembersMap().remove(member.getUser().getIdLong());
66        api.getEventManager()
67            .handle(
68                new GuildVoiceLeaveEvent(
69                    api, responseNumber,
70                    member, channel));
71      }
72  
73      // The user is not in a different guild that we share
74      // The user also is not a friend of this account in the case that the logged in account is a
75      // client account.
76      if (userId != api.getSelfUser().getIdLong() // don't remove selfUser from cache
77          && api.getGuildMap()
78              .valueCollection()
79              .stream()
80              .noneMatch(g -> ((GuildImpl) g).getMembersMap().containsKey(userId))) {
81        UserImpl user = (UserImpl) api.getUserMap().remove(userId);
82        if (user.hasPrivateChannel()) {
83          PrivateChannelImpl priv = (PrivateChannelImpl) user.getPrivateChannel();
84          user.setFake(true);
85          priv.setFake(true);
86          api.getFakeUserMap().put(user.getIdLong(), user);
87          api.getFakePrivateChannelMap().put(priv.getIdLong(), priv);
88        }
89        api.getEventCache().clear(EventCache.Type.USER, userId);
90      }
91      api.getEventManager()
92          .handle(
93              new GuildMemberLeaveEvent(
94                  api, responseNumber,
95                  guild, member));
96      return null;
97    }
98  }
