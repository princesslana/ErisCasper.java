37  public class ReadyHandler extends SocketHandler {
38    private final TLongSet incompleteGuilds = new TLongHashSet();
39    private final TLongSet acknowledgedGuilds = new TLongHashSet();
40    private final TLongSet unavailableGuilds = new TLongHashSet();
41    private final TLongSet guildsRequiringChunking = new TLongHashSet();
42    private final TLongSet guildsRequiringSyncing = new TLongHashSet();
43  
44    public ReadyHandler(JDAImpl api) {
45      super(api);
46    }
47  
48    @Override
49    protected Long handleInternally(JSONObject content) {
50      EntityBuilder builder = api.getEntityBuilder();
51  
52      // Core
53      JSONArray guilds = content.getJSONArray("guilds");
54      JSONObject selfJson = content.getJSONObject("user");
55  
56      builder.createSelfUser(selfJson);
57  
58      if (api.getAccountType() == AccountType.CLIENT && !content.isNull("user_settings")) {
59        // handle user settings
60        JSONObject userSettingsJson = content.getJSONObject("user_settings");
61        UserSettingsImpl userSettingsObj = (UserSettingsImpl) api.asClient().getSettings();
62        userSettingsObj
63            // TODO: set all information and handle updates
64            .setStatus(
65            userSettingsJson.isNull("status")
66                ? OnlineStatus.ONLINE
67                : OnlineStatus.fromKey(userSettingsJson.getString("status")));
68        // update presence information unless the status is ONLINE
69        if (userSettingsObj.getStatus() != OnlineStatus.ONLINE)
70          ((PresenceImpl) api.getPresence()).setCacheStatus(userSettingsObj.getStatus());
71      }
72  
73      // Keep a list of all guilds in incompleteGuilds that need to be setup (GuildMemberChunk /
74      // GuildSync)
75      // Send all guilds to the EntityBuilder's first pass to setup caching for when GUILD_CREATE
76      // comes
77      // or, for Client accounts, to start the setup process (since we already have guild info)
78      // Callback points to guildSetupComplete so that when MemberChunking and GuildSync processes are
79      // done, we can
80      // "check off" the completed guild from the set of guilds in incompleteGuilds.
81  
82      for (int i = 0; i < guilds.length(); i++) {
83        JSONObject guild = guilds.getJSONObject(i);
84        incompleteGuilds.add(guild.getLong("id"));
85      }
86  
87      // We use two different for-loops here so that we cache all of the ids before sending them off
88      // to the EntityBuilder
89      //  due to the checks in checkIfReadyToSendRequests and guildSetupComplete triggering too soon
90      // otherwise.
91      // Specifically: incompleteGuilds.size() == acknowledgedGuilds.size() and
92      //  incompleteGuilds.size() == unavailableGuilds.size() respectively.
93  
94      for (int i = 0; i < guilds.length(); i++) {
95        JSONObject guild = guilds.getJSONObject(i);
96  
97        // If a Guild isn't unavailable, then it is possible that we were given all information
98        // needed to fully load the guild. In this case, we provide the method `guildSetupComplete`
99        // as the secondPassCallback so it can immediately be called to signify that the provided
100       // guild
101       // is loaded and ready to go.
102       // If a Guild is unavailable it won't have the information needed, so we pass null as the
103       // secondPassCallback
104       // for now and wait for the GUILD_CREATE event to give us the required information.
105       if (guild.has("unavailable") && guild.getBoolean("unavailable"))
106         builder.createGuildFirstPass(guild, null);
107       else builder.createGuildFirstPass(guild, this::guildSetupComplete);
108     }
109 
110     if (guilds.length() == 0) guildLoadComplete(content);
111 
112     return null;
113   }
114 
115   public void guildLoadComplete(JSONObject content) {
116     api.getClient().setChunkingAndSyncing(false);
117     EntityBuilder builder = api.getEntityBuilder();
118     JSONArray privateChannels = content.getJSONArray("private_channels");
119 
120     if (api.getAccountType() == AccountType.CLIENT) {
121       JSONArray relationships = content.getJSONArray("relationships");
122       JSONArray presences = content.getJSONArray("presences");
123       JSONObject notes = content.getJSONObject("notes");
124       JSONArray readstates = content.has("read_state") ? content.getJSONArray("read_state") : null;
125       JSONArray guildSettings =
126           content.has("user_guild_settings") ? content.getJSONArray("user_guild_settings") : null;
127 
128       for (int i = 0; i < relationships.length(); i++) {
129         JSONObject relationship = relationships.getJSONObject(i);
130         Relationship r = builder.createRelationship(relationship);
131         if (r == null)
132           JDAImpl.LOG.error(
133               "Provided relationship in READY with an unknown type! JSON: {}", relationship);
134       }
135 
136       for (int i = 0; i < presences.length(); i++) {
137         JSONObject presence = presences.getJSONObject(i);
138         String userId = presence.getJSONObject("user").getString("id");
139         FriendImpl friend = (FriendImpl) api.asClient().getFriendById(userId);
140         if (friend == null)
141           WebSocketClient.LOG.warn(
142               "Received a presence in the Presences array in READY that did not correspond to a cached Friend! JSON: {}",
143               presence);
144         else builder.createPresence(friend, presence);
145       }
146     }
147 
148     for (int i = 0; i < privateChannels.length(); i++) {
149       JSONObject chan = privateChannels.getJSONObject(i);
150       ChannelType type = ChannelType.fromId(chan.getInt("type"));
151 
152       switch (type) {
153         case PRIVATE:
154           builder.createPrivateChannel(chan);
155           break;
156         case GROUP:
157           builder.createGroup(chan);
158           break;
159         default:
160           WebSocketClient.LOG.warn(
161               "Received a Channel in the priv_channels array in READY of an unknown type! JSON: {}",
162               type);
163       }
164     }
165 
166     api.getClient().ready();
167   }

169   public void acknowledgeGuild(
170       Guild guild, boolean available, boolean requiresChunking, boolean requiresSync) {
171     acknowledgedGuilds.add(guild.getIdLong());
172     if (available) {
173       // We remove from unavailable guilds because it is possible that we were told it was
174       // unavailable, but
175       // during a long READY load it could have become available and was sent to us.
176       unavailableGuilds.remove(guild.getIdLong());
177       if (requiresChunking) guildsRequiringChunking.add(guild.getIdLong());
178       if (requiresSync) guildsRequiringSyncing.add(guild.getIdLong());
179     } else unavailableGuilds.add(guild.getIdLong());
180 
181     checkIfReadyToSendRequests();
182   }
183 
184   public void guildSetupComplete(Guild guild) {
185     if (!incompleteGuilds.remove(guild.getIdLong()))
186       WebSocketClient.LOG.error(
187           "Completed the setup for Guild: {} without matching id in ReadyHandler cache", guild);
188     if (incompleteGuilds.size() == unavailableGuilds.size())
189       guildLoadComplete(allContent.getJSONObject("d"));
190     else checkIfReadyToSendRequests();
191   }
192 
193   public void clearCache() {
194     incompleteGuilds.clear();
195     acknowledgedGuilds.clear();
196     unavailableGuilds.clear();
197     guildsRequiringChunking.clear();
198     guildsRequiringSyncing.clear();
199   }
200 
201   private void checkIfReadyToSendRequests() {
202     if (acknowledgedGuilds.size() == incompleteGuilds.size()) {
203       api.getClient().setChunkingAndSyncing(true);
204       if (api.getAccountType() == AccountType.CLIENT) sendGuildSyncRequests();
205       sendMemberChunkRequests();
206     }
207   }
208 
209   private void sendGuildSyncRequests() {
210     if (guildsRequiringSyncing.isEmpty()) return;
211 
212     JSONArray guildIds = new JSONArray();
213 
214     for (TLongIterator it = guildsRequiringSyncing.iterator(); it.hasNext(); ) {
215       guildIds.put(it.next());
216 
217       // We can only request 50 guilds in a single request, so after we've reached 50, send them
218       // and reset the
219       if (guildIds.length() == 50) {
220         api.getClient()
221             .chunkOrSyncRequest(
222                 new JSONObject().put("op", WebSocketCode.GUILD_SYNC).put("d", guildIds));
223         guildIds = new JSONArray();
224       }
225     }
226 
227     // Send the remaining guilds that need to be sent
228     if (guildIds.length() > 0) {
229       api.getClient()
230           .chunkOrSyncRequest(
231               new JSONObject().put("op", WebSocketCode.GUILD_SYNC).put("d", guildIds));
232     }
233     guildsRequiringSyncing.clear();
234   }
235 
236   private void sendMemberChunkRequests() {
237     if (guildsRequiringChunking.isEmpty()) return;
238 
239     JSONArray guildIds = new JSONArray();
240     for (TLongIterator it = guildsRequiringChunking.iterator(); it.hasNext(); ) {
241       guildIds.put(it.next());
242 
243       // We can only request 50 guilds in a single request, so after we've reached 50, send them
244       // and reset the
245       if (guildIds.length() == 50) {
246         api.getClient()
247             .chunkOrSyncRequest(
248                 new JSONObject()
249                     .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
250                     .put(
251                         "d",
252                         new JSONObject()
253                             .put("guild_id", guildIds)
254                             .put("query", "")
255                             .put("limit", 0)));
256         guildIds = new JSONArray();
257       }
258     }
259 
260     // Send the remaining guilds that need to be sent
261     if (guildIds.length() > 0) {
262       api.getClient()
263           .chunkOrSyncRequest(
264               new JSONObject()
265                   .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
266                   .put(
267                       "d",
268                       new JSONObject().put("guild_id", guildIds).put("query", "").put("limit", 0)));
269     }
270     guildsRequiringChunking.clear();
271   }
