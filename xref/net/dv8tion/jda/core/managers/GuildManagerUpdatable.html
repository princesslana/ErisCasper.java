36  /**
37   * An {@link #update() updatable} manager that allows to modify guild settings like the {@link
38   * #getNameField() name} or the {@link #getSplashField() splash}.
39   *
40   * <p>This manager allows to modify multiple fields at once by getting the {@link
41   * net.dv8tion.jda.core.managers.fields.GuildField GuildField} for specific properties and setting
42   * or resetting their values; followed by a call of {@link #update()}!
43   *
44   * <p>The {@link net.dv8tion.jda.core.managers.GuildManager GuildManager} implementation simplifies
45   * this process by giving simple setters that return the {@link #update() update} {@link
46   * net.dv8tion.jda.core.requests.RestAction RestAction}
47   *
48   * <p><b>Note</b>: To {@link #update() update} this manager the currently logged in account requires
49   * the Permission {@link net.dv8tion.jda.core.Permission#MANAGE_SERVER MANAGE_SERVER}
50   *
51   * <p><b>To use moderation abilities like creating Roles or banning Members use {@link
52   * net.dv8tion.jda.core.managers.GuildController GuildController}</b>
53   */

396   /**
397    * Creates a new {@link net.dv8tion.jda.core.requests.RestAction RestAction} instance that will
398    * apply <b>all</b> changes that have been made to this manager instance. <br>
399    * If no changes have been made this will simply return {@link
400    * net.dv8tion.jda.core.requests.RestAction.EmptyRestAction EmptyRestAction}.
401    *
402    * <p>Before applying new changes it is recommended to call {@link #reset()} to reset previous
403    * changes. <br>
404    * This is automatically called if this method returns successfully.
405    *
406    * <p>Possible {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponses} for this update
407    * include the following:
408    *
409    * <ul>
410    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_GUILD UNKNOWN_GUILD} <br>
411    *       If the Guild was deleted before finishing the task
412    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_PERMISSIONS
413    *       MISSING_PERMISSIONS} <br>
414    *       If the currently logged in account loses the {@link
415    *       net.dv8tion.jda.core.Permission#MANAGE_SERVER MANAGE_SERVER Permission} before finishing
416    *       the task
417    * </ul>
418    *
419    * @throws net.dv8tion.jda.core.exceptions.InsufficientPermissionException If the currently logged
420    *     in account does not have the Permission {@link
421    *     net.dv8tion.jda.core.Permission#MANAGE_SERVER MANAGE_SERVER} in the underlying {@link
422    *     net.dv8tion.jda.core.entities.Guild Guild}
423    * @throws net.dv8tion.jda.core.exceptions.GuildUnavailableException If the Guild is temporarily
424    *     not {@link net.dv8tion.jda.core.entities.Guild#isAvailable() available}
425    * @return {@link net.dv8tion.jda.core.requests.restaction.AuditableRestAction
426    *     AuditableRestAction} <br>
427    *     Applies all changes that have been made in a single api-call.
428    */
429   public AuditableRestAction<Void> update() {
430     checkAvailable();
431     checkPermission(Permission.MANAGE_SERVER);
432 
433     if (!needToUpdate()) return new AuditableRestAction.EmptyRestAction<>(getJDA(), null);
434 
435     JSONObject body = new JSONObject().put("name", guild.getName());
436     if (name.shouldUpdate()) body.put("name", name.getValue());
437     if (region.shouldUpdate()) body.put("region", region.getValue().getKey());
438     if (timeout.shouldUpdate()) body.put("afk_timeout", timeout.getValue().getSeconds());
439     if (icon.shouldUpdate())
440       body.put("icon", icon.getValue() == null ? JSONObject.NULL : icon.getValue().getEncoding());
441     if (splash.shouldUpdate())
442       body.put(
443           "splash", splash.getValue() == null ? JSONObject.NULL : splash.getValue().getEncoding());
444     if (afkChannel.shouldUpdate())
445       body.put(
446           "afk_channel_id",
447           afkChannel.getValue() == null ? JSONObject.NULL : afkChannel.getValue().getId());
448     if (systemChannel.shouldUpdate())
449       body.put(
450           "system_channel_id",
451           systemChannel.getValue() == null ? JSONObject.NULL : systemChannel.getValue().getId());
452     if (verificationLevel.shouldUpdate())
453       body.put("verification_level", verificationLevel.getValue().getKey());
454     if (defaultNotificationLevel.shouldUpdate())
455       body.put("default_notification_level", defaultNotificationLevel.getValue().getKey());
456     if (mfaLevel.shouldUpdate()) body.put("mfa_level", mfaLevel.getValue().getKey());
457     if (explicitContentLevel.shouldUpdate())
458       body.put("explicit_content_filter", explicitContentLevel.getValue().getKey());
459 
460     reset(); // now that we've built our JSON object, reset the manager back to the non-modified
461     // state
462     Route.CompiledRoute route = Route.Guilds.MODIFY_GUILD.compile(guild.getId());
463     return new AuditableRestAction<Void>(guild.getJDA(), route, body) {
464       @Override
465       protected void handleResponse(Response response, Request<Void> request) {
466         if (response.isOk()) request.onSuccess(null);
467         else request.onFailure(response);
468       }
469     };
470   }

474         || region.shouldUpdate()
475         || timeout.shouldUpdate()
476         || icon.shouldUpdate()
477         || splash.shouldUpdate()
478         || afkChannel.shouldUpdate()
479         || systemChannel.shouldUpdate()
480         || verificationLevel.shouldUpdate()
481         || defaultNotificationLevel.shouldUpdate()
482         || mfaLevel.shouldUpdate()
