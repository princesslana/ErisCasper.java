17  package net.dv8tion.jda.core.utils;
18  
19  import java.io.*;
20  
21  public class IOUtil {
22  
23    /**
24     * Used as an alternate to Java's nio Files.readAllBytes.
25     *
26     * <p>This customized version for File is provide (instead of just using {@link
27     * #readFully(java.io.InputStream)} with a FileInputStream) because with a File we can determine
28     * the total size of the array and do not need to have a buffer. This results in a memory
29     * footprint that is half the size of {@link #readFully(java.io.InputStream)}
30     *
31     * <p>Code provided from <a href="http://stackoverflow.com/a/6276139">Stackoverflow</a>
32     *
33     * @param file The file from which we should retrieve the bytes from
34     * @throws java.io.IOException Thrown if there is a problem while reading the file.
35     * @return A byte[] containing all of the file's data
36     */
37    public static byte[] readFully(File file) throws IOException {
38      Checks.notNull(file, "File");
39      Checks.check(file.exists(), "Provided file does not exist!");
40  
41      try (InputStream is = new FileInputStream(file)) {
42        // Get the size of the file
43        long length = file.length();
44  
45        // You cannot create an array using a long type.
46        // It needs to be an int type.
47        // Before converting to an int type, check
48        // to ensure that file is not larger than Integer.MAX_VALUE.
49        if (length > Integer.MAX_VALUE) {
50          throw new IOException(
51              "Cannot read the file into memory completely due to it being too large!");
52          // File is too large
53        }
54  
55        // Create the byte array to hold the data
56        byte[] bytes = new byte[(int) length];
57  
58        // Read in the bytes
59        int offset = 0;
60        int numRead = 0;
61        while (offset < bytes.length
62            && (numRead = is.read(bytes, offset, bytes.length - offset)) >= 0) {
63          offset += numRead;
64        }
65  
66        // Ensure all the bytes have been read in
67        if (offset < bytes.length) {
68          throw new IOException("Could not completely read file " + file.getName());
69        }
70  
71        // Close the input stream and return bytes
72        is.close();
73        return bytes;
74      }
75    }
76  
77    /**
78     * Provided as a simple way to fully read an InputStream into a byte[].
79     *
80     * <p>This method will block until the InputStream has been fully read, so if you provide an
81     * InputStream that is non-finite, you're gonna have a bad time.
82     *
83     * @param stream The Stream to be read.
84     * @throws IOException If the first byte cannot be read for any reason other than the end of the
85     *     file, if the input stream has been closed, or if some other I/O error occurs.
86     * @return A byte[] containing all of the data provided by the InputStream
87     */
88    public static byte[] readFully(InputStream stream) throws IOException {
89      Checks.notNull(stream, "InputStream");
90  
91      byte[] buffer = new byte[1024];
92      try (ByteArrayOutputStream bos = new ByteArrayOutputStream()) {
93        int readAmount = 0;
94        while ((readAmount = stream.read(buffer)) != -1) {
95          bos.write(buffer, 0, readAmount);
96        }
97        return bos.toByteArray();
98      }
99    }
