27    /**
28     * Checks if one given Member can interact with a 2nd given Member - in a permission sense
29     * (kick/ban/modify perms). This only checks the Role-Position and does not check the actual
30     * permission (kick/ban/manage_role/...)
31     *
32     * @param issuer The member that tries to interact with 2nd member
33     * @param target The member that is the target of the interaction
34     * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
35     *     provided entities are not from the same guild
36     * @return True, if issuer can interact with target in guild
37     */
38    public static boolean canInteract(Member issuer, Member target) {
39      Checks.notNull(issuer, "Issuer Member");
40      Checks.notNull(target, "Target Member");
41  
42      Guild guild = issuer.getGuild();
43      if (!guild.equals(target.getGuild()))
44        throw new IllegalArgumentException(
45            "Provided members must both be Member objects of the same Guild!");
46      if (guild.getOwner().equals(issuer)) return true;
47      if (guild.getOwner().equals(target)) return false;
48      List<Role> issuerRoles = issuer.getRoles();
49      List<Role> targetRoles = target.getRoles();
50      return !issuerRoles.isEmpty()
51          && (targetRoles.isEmpty() || canInteract(issuerRoles.get(0), targetRoles.get(0)));
52    }

55     * Checks if a given Member can interact with a given Role - in a permission sense
56     * (kick/ban/modify perms). This only checks the Role-Position and does not check the actual
57     * permission (kick/ban/manage_role/...)
58     *
59     * @param issuer The member that tries to interact with the role
60     * @param target The role that is the target of the interaction
61     * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
62     *     provided entities are not from the same guild
63     * @return True, if issuer can interact with target
64     */
65    public static boolean canInteract(Member issuer, Role target) {
66      Checks.notNull(issuer, "Issuer Member");
67      Checks.notNull(target, "Target Role");
68  
69      Guild guild = issuer.getGuild();
70      if (!guild.equals(target.getGuild()))
71        throw new IllegalArgumentException(
72            "Provided Member issuer and Role target must be from the same Guild!");
73      if (guild.getOwner().equals(issuer)) return true;
74      List<Role> issuerRoles = issuer.getRoles();
75      return !issuerRoles.isEmpty() && canInteract(issuerRoles.get(0), target);
76    }

98    /**
99     * Check whether the provided {@link net.dv8tion.jda.core.entities.Member Member} can use the
100    * specified {@link net.dv8tion.jda.core.entities.Emote Emote}.
101    *
102    * <p>If the specified Member is not in the emote's guild or the emote provided is fake this will
103    * return false. Otherwise it will check if the emote is restricted to any roles and if that is
104    * the case if the Member has one of these.
105    *
106    * <p>In the case of an {@link net.dv8tion.jda.core.entities.Emote#isAnimated() animated} Emote,
107    * this will check if the issuer is the currently logged in account, and then check if the account
108    * has {@link net.dv8tion.jda.core.entities.SelfUser#isNitro() nitro}, and return false if it
109    * doesn't. <br>
110    * For other accounts, this method will not take into account whether the emote is animated, as
111    * there is no real way to check if the Member can interact with them. <br>
112    * <b>Note</b>: This is not checking if the issuer owns the Guild or not.
113    *
114    * @param issuer The member that tries to interact with the Emote
115    * @param emote The emote that is the target interaction
116    * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
117    *     provided entities are not from the same guild
118    * @return True, if the issuer can interact with the emote
119    */
120   public static boolean canInteract(Member issuer, Emote emote) {
121     Checks.notNull(issuer, "Issuer Member");
122     Checks.notNull(emote, "Target Emote");
123 
124     if (!issuer.getGuild().equals(emote.getGuild()))
125       throw new IllegalArgumentException("The issuer and target are not in the same Guild");
126 
127     // We don't need to check based on the fact it is animated if it's a BOT account
128     // because BOT accounts cannot have nitro, and have access to animated Emotes naturally.
129     if (emote.isAnimated() && !issuer.getUser().isBot()) {
130       // This is a currently logged in client, meaning we can check if they have nitro or not.
131       // If this isn't the currently logged in account, we just check it like a normal emote,
132       // since there is no way to verify if they have nitro or not.
133       if (issuer.getUser() instanceof SelfUser) {
134         // If they don't have nitro, we immediately return
135         // false, otherwise we proceed with the remaining checks.
136         if (!((SelfUser) issuer.getUser()).isNitro()) return false;
137       }
138     }
139 
140     return (emote.getRoles().isEmpty() // Emote restricted to roles -> check if the issuer has them
141         || CollectionUtils.containsAny(issuer.getRoles(), emote.getRoles()));
142   }

144   /**
145    * Checks whether the specified {@link net.dv8tion.jda.core.entities.Emote Emote} can be used by
146    * the provided {@link net.dv8tion.jda.core.entities.User User} in the {@link
147    * net.dv8tion.jda.core.entities.MessageChannel MessageChannel}.
148    *
149    * @param issuer The user that tries to interact with the Emote
150    * @param emote The emote that is the target interaction
151    * @param channel The MessageChannel this emote should be interacted within
152    * @param botOverride Whether bots can use non-managed emotes in other guilds
153    * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
154    *     provided entities are not from the same guild
155    * @return True, if the issuer can interact with the emote within the specified MessageChannel
156    */
157   public static boolean canInteract(
158       User issuer, Emote emote, MessageChannel channel, boolean botOverride) {
159     Checks.notNull(issuer, "Issuer Member");
160     Checks.notNull(emote, "Target Emote");
161     Checks.notNull(channel, "Target Channel");
162 
163     if (emote.isFake() || !emote.getGuild().isMember(issuer))
164       return false; // cannot use an emote if you're not in its guild
165     Member member = emote.getGuild().getMemberById(issuer.getIdLong());
166     if (!canInteract(member, emote)) return false;
167     // external means it is available outside of its own guild - works for bots or if its managed
168     // currently we cannot check whether other users have nitro, we assume no here
169     final boolean external =
170         emote.isManaged() || (issuer.isBot() && botOverride) || isNitro(issuer);
171     switch (channel.getType()) {
172       case TEXT:
173         TextChannel text = (TextChannel) channel;
174         member = text.getGuild().getMemberById(issuer.getIdLong());
175         return emote.getGuild().equals(text.getGuild()) // within the same guild
176             || (external
177                 && member.hasPermission(text, Permission.MESSAGE_EXT_EMOJI)); // in different guild
178       default:
179         return external; // In Group or Private it only needs to be external
180     }
181   }

185   }

219    * @return True - if the {@link net.dv8tion.jda.core.entities.Member Member} effectively has the
220    *     specified {@link net.dv8tion.jda.core.Permission Permissions}.
221    */
222   public static boolean checkPermission(Member member, Permission... permissions) {
223     Checks.notNull(member, "Member");
224     Checks.notNull(permissions, "Permissions");
225 
226     long effectivePerms = getEffectivePermission(member);
227     return isApplied(effectivePerms, Permission.ADMINISTRATOR.getRawValue())
228         || isApplied(effectivePerms, Permission.getRaw(permissions));
229   }

272   /**
273    * Gets the {@code long} representation of the effective permissions allowed for this {@link
274    * net.dv8tion.jda.core.entities.Member Member} in this {@link net.dv8tion.jda.core.entities.Guild
275    * Guild}. This can be used in conjunction with {@link
276    * net.dv8tion.jda.core.Permission#getPermissions(long) Permission.getPermissions(int)} to easily
277    * get a list of all {@link net.dv8tion.jda.core.Permission Permissions} that this member has in
278    * this {@link net.dv8tion.jda.core.entities.Guild Guild}.
279    *
280    * <p><b>This only returns the Guild-level permissions!</b>
281    *
282    * @param member The {@link net.dv8tion.jda.core.entities.Member Member} whose permissions are
283    *     being checked.
284    * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
285    *     provided entities are not from the same guild
286    * @return The {@code long} representation of the literal permissions that this {@link
287    *     net.dv8tion.jda.core.entities.Member Member} has in this {@link
288    *     net.dv8tion.jda.core.entities.Guild Guild}.
289    */
290   public static long getEffectivePermission(Member member) {
291     Checks.notNull(member, "Member");
292 
293     if (member.isOwner()) return Permission.ALL_PERMISSIONS;
294     // Default to binary OR of all global permissions in this guild
295     long permission = member.getGuild().getPublicRole().getPermissionsRaw();
296     for (Role role : member.getRoles()) {
297       permission |= role.getPermissionsRaw();
298       if (isApplied(permission, Permission.ADMINISTRATOR.getRawValue()))
299         return Permission.ALL_PERMISSIONS;
300     }
301 
302     return permission;
303   }

306    * Gets the {@code long} representation of the effective permissions allowed for this {@link
307    * net.dv8tion.jda.core.entities.Member Member} in this {@link
308    * net.dv8tion.jda.core.entities.Channel Channel}. This can be used in conjunction with {@link
309    * net.dv8tion.jda.core.Permission#getPermissions(long) Permission.getPermissions(long)} to easily
310    * get a list of all {@link net.dv8tion.jda.core.Permission Permissions} that this member can use
311    * in this {@link net.dv8tion.jda.core.entities.Channel Channel}. <br>
312    * This functions very similarly to how {@link
313    * net.dv8tion.jda.core.entities.Role#getPermissionsRaw() Role.getPermissionsRaw()}.
314    *
315    * @param channel The {@link net.dv8tion.jda.core.entities.Channel Channel} being checked.
316    * @param member The {@link net.dv8tion.jda.core.entities.Member Member} whose permissions are
317    *     being checked.
318    * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
319    *     provided entities are not from the same guild
320    * @return The {@code long} representation of the effective permissions that this {@link
321    *     net.dv8tion.jda.core.entities.Member Member} has in this {@link
322    *     net.dv8tion.jda.core.entities.Channel Channel}.
323    */
324   public static long getEffectivePermission(Channel channel, Member member) {
325     Checks.notNull(channel, "Channel");
326     Checks.notNull(member, "Member");
327 
328     Checks.check(
329         channel.getGuild().equals(member.getGuild()),
330         "Provided channel and provided member are not of the same guild!");
331 
332     if (member.isOwner()) {
333       // Owner effectively has all permissions
334       return Permission.ALL_PERMISSIONS;
335     }
336 
337     long permission = getEffectivePermission(member);
338     final long admin = Permission.ADMINISTRATOR.getRawValue();
339     if (isApplied(permission, admin)) return Permission.ALL_PERMISSIONS;
340 
341     AtomicLong allow = new AtomicLong(0);
342     AtomicLong deny = new AtomicLong(0);
343     getExplicitOverrides(channel, member, allow, deny);
344     permission = apply(permission, allow.get(), deny.get());
345     final long viewChannel = Permission.VIEW_CHANNEL.getRawValue();
346 
347     // When the permission to view the channel is not applied it is not granted
348     // This means that we have no access to this channel at all
349     return isApplied(permission, viewChannel) ? permission : 0;
350     /*
351     // currently discord doesn't implicitly grant permissions that the user can grant others
352     // so instead the user has to explicitly make an override to grant them the permission in order to be granted that permission
353     // yes this makes no sense but what can i do, the devs don't like changing things apparently...
354     // I've been told half a year ago this would be changed but nothing happens
355     // so instead I'll just bend over for them so people get "correct" permission checks...
356     //
357     // only time will tell if something happens and I can finally re-implement this section wew
358     final long managePerms = Permission.MANAGE_PERMISSIONS.getRawValue();
359     final long manageChannel = Permission.MANAGE_CHANNEL.getRawValue();
360     if ((permission & (managePerms | manageChannel)) != 0)
361     {
362         // In channels, MANAGE_CHANNEL and MANAGE_PERMISSIONS grant full text/voice permissions
363         permission |= Permission.ALL_TEXT_PERMISSIONS | Permission.ALL_VOICE_PERMISSIONS;
364     }
365     */
366   }

368   /**
369    * Gets the {@code long} representation of the effective permissions allowed for this {@link
370    * net.dv8tion.jda.core.entities.Role Role} in this {@link net.dv8tion.jda.core.entities.Channel
371    * Channel}. This can be used in conjunction with {@link
372    * net.dv8tion.jda.core.Permission#getPermissions(long) Permission.getPermissions(long)} to easily
373    * get a list of all {@link net.dv8tion.jda.core.Permission Permissions} that this role can use in
374    * this {@link net.dv8tion.jda.core.entities.Channel Channel}.
375    *
376    * @param channel The {@link net.dv8tion.jda.core.entities.Channel Channel} in which permissions
377    *     are being checked.
378    * @param role The {@link net.dv8tion.jda.core.entities.Role Role} whose permissions are being
379    *     checked.
380    * @throws IllegalArgumentException if any of the provided parameters is {@code null} or the
381    *     provided entities are not from the same guild
382    * @return The {@code long} representation of the effective permissions that this {@link
383    *     net.dv8tion.jda.core.entities.Role Role} has in this {@link
384    *     net.dv8tion.jda.core.entities.Channel Channel}
385    */
386   public static long getEffectivePermission(Channel channel, Role role) {
387     Checks.notNull(channel, "Channel");
388     Checks.notNull(role, "Role");
389 
390     Guild guild = channel.getGuild();
391     if (!guild.equals(role.getGuild()))
392       throw new IllegalArgumentException("Provided channel and role are not of the same guild!");
393 
394     long permissions = role.getPermissionsRaw() | guild.getPublicRole().getPermissionsRaw();
395 
396     PermissionOverride publicOverride = channel.getPermissionOverride(guild.getPublicRole());
397     PermissionOverride roleOverride = channel.getPermissionOverride(role);
398 
399     if (publicOverride != null) {
400       permissions &= ~publicOverride.getDeniedRaw();
401       permissions |= publicOverride.getAllowedRaw();
402     }
403 
404     if (roleOverride != null) {
405       permissions &= ~roleOverride.getDeniedRaw();
406       permissions |= roleOverride.getAllowedRaw();
407     }
408 
409     return permissions;
410   }

413    * Retrieves the explicit permissions of the specified {@link net.dv8tion.jda.core.entities.Member
414    * Member} in its hosting {@link net.dv8tion.jda.core.entities.Guild Guild}. <br>
415    * This method does not calculate the owner in.
416    *
417    * <p>All permissions returned are explicitly granted to this Member via its {@link
418    * net.dv8tion.jda.core.entities.Role Roles}. <br>
419    * Permissions like {@link net.dv8tion.jda.core.Permission#ADMINISTRATOR Permission.ADMINISTRATOR}

421    *

424    * @throws IllegalArgumentException If the specified member is {@code null}
425    * @return Primitive (unsigned) long value with the implicit permissions of the specified member
426    * @since 3.1
427    */
428   public static long getExplicitPermission(Member member) {
429     Checks.notNull(member, "Member");
430 
431     final Guild guild = member.getGuild();
432     long permission = guild.getPublicRole().getPermissionsRaw();
433 
434     for (Role role : member.getRoles()) permission |= role.getPermissionsRaw();
435 
436     return permission;
437   }

441    * Member} in its hosting {@link net.dv8tion.jda.core.entities.Guild Guild} and specific {@link
442    * net.dv8tion.jda.core.entities.Channel Channel}. <br>
443    * This method does not calculate the owner in. <b>Allowed permissions override denied permissions

445    *
446    * <p>All permissions returned are explicitly granted to this Member via its {@link
447    * net.dv8tion.jda.core.entities.Role Roles}. <br>
448    * Permissions like {@link net.dv8tion.jda.core.Permission#ADMINISTRATOR Permission.ADMINISTRATOR}
449    * do not grant other permissions in this value.
450    *
451    * <p>This factor in all {@link net.dv8tion.jda.core.entities.PermissionOverride
452    * PermissionOverrides} that affect this member and only grants the ones that are explicitly
453    * given.
454    *
455    * @param channel The target channel of which to check {@link
456    *     net.dv8tion.jda.core.entities.PermissionOverride PermissionOverrides}
457    * @param member The non-null {@link net.dv8tion.jda.core.entities.Member Member} for which to get
458    *     implicit permissions
459    * @throws IllegalArgumentException If any of the arguments is {@code null} or the specified
460    *     entities are not from the same {@link net.dv8tion.jda.core.entities.Guild Guild}
461    * @return Primitive (unsigned) long value with the implicit permissions of the specified member
462    *     in the specified channel
463    * @since 3.1
464    */

466     Checks.notNull(channel, "Channel");
467     Checks.notNull(member, "Member");
468 
469     final Guild guild = member.getGuild();
470     checkGuild(channel.getGuild(), guild, "Member");
471 
472     long permission = getExplicitPermission(member);
473 
474     AtomicLong allow = new AtomicLong(0);
475     AtomicLong deny = new AtomicLong(0);
476 
477     // populates allow/deny
478     getExplicitOverrides(channel, member, allow, deny);
479 
480     return apply(permission, allow.get(), deny.get());
481   }

496    *
497    * @param channel The target channel of which to check {@link
498    *     net.dv8tion.jda.core.entities.PermissionOverride PermissionOverrides}
499    * @param role The non-null {@link net.dv8tion.jda.core.entities.Role Role} for which to get
500    *     implicit permissions
501    * @throws IllegalArgumentException If any of the arguments is {@code null} or the specified
502    *     entities are not from the same {@link net.dv8tion.jda.core.entities.Guild Guild}
503    * @return Primitive (unsigned) long value with the implicit permissions of the specified role in
504    *     the specified channel
505    * @since 3.1
506    */
507   public static long getExplicitPermission(Channel channel, Role role) {
508     Checks.notNull(channel, "Channel");
509     Checks.notNull(role, "Role");
510 
511     final Guild guild = role.getGuild();
512     checkGuild(channel.getGuild(), guild, "Role");
513 
514     long permission = role.getPermissionsRaw() | guild.getPublicRole().getPermissionsRaw();
515     PermissionOverride override = channel.getPermissionOverride(guild.getPublicRole());
516     if (override != null)
517       permission = apply(permission, override.getAllowedRaw(), override.getDeniedRaw());
518     if (role.isPublicRole()) return permission;
519 
520     override = channel.getPermissionOverride(role);
521 
522     return override == null
523         ? permission
524         : apply(permission, override.getAllowedRaw(), override.getDeniedRaw());
525   }
526 
527   /**
528    * Pushes all deny/allow values to the specified BiConsumer <br>
529    * First parameter is allow, second is deny
530    */
531   private static void getExplicitOverrides(
532       Channel channel, Member member, AtomicLong allow, AtomicLong deny) {
533     PermissionOverride override = channel.getPermissionOverride(member.getGuild().getPublicRole());
534     long allowRaw = 0;
535     long denyRaw = 0;
536     if (override != null) {
537       denyRaw = override.getDeniedRaw();
538       allowRaw = override.getAllowedRaw();
539     }
540 
541     long allowRole = 0;
542     long denyRole = 0;
543     // create temporary bit containers for role cascade
544     for (Role role : member.getRoles()) {
545       override = channel.getPermissionOverride(role);
546       if (override != null) {
547         // important to update role cascade not others
548         denyRole |= override.getDeniedRaw();
549         allowRole |= override.getAllowedRaw();
550       }
551     }
552     // Override the raw values of public role then apply role cascade
553     allowRaw = (allowRaw & ~denyRole) | allowRole;
554     denyRaw = (denyRaw & ~allowRole) | denyRole;
555 
556     override = channel.getPermissionOverride(member);
557     if (override != null) {
558       // finally override the role cascade with member overrides
559       final long oDeny = override.getDeniedRaw();
560       final long oAllow = override.getAllowedRaw();
561       allowRaw = (allowRaw & ~oDeny) | oAllow;
562       denyRaw = (denyRaw & ~oAllow) | oDeny;
563       // this time we need to exclude new allowed bits from old denied ones and OR the new denied
564       // bits as final overrides
565     }
566     // set as resulting values
567     allow.set(allowRaw);
568     deny.set(denyRaw);
569   }
570 
571   /*
572    * Check whether the specified permission is applied in the bits
573    */
574   private static boolean isApplied(long permissions, long perms) {
575     return (permissions & perms) == perms;
576   }
577 
578   private static long apply(long permission, long allow, long deny) {
579     permission &= ~deny; // Deny everything that the cascade of roles denied.
580     permission |= allow; // Allow all the things that the cascade of roles allowed
581     // The allowed bits override the denied ones!
582     return permission;
583   }

585   private static void checkGuild(Guild o1, Guild o2, String name) {
586     Checks.check(
587         o1.equals(o2),
588         "Specified %s is not in the same guild! (%s / %s)",
589         name,
590         o1.toString(),
591         o2.toString());
592   }
