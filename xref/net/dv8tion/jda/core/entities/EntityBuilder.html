70    public SelfUser createSelfUser(JSONObject self) {
71      SelfUserImpl selfUser = ((SelfUserImpl) api.getSelfUser());
72      if (selfUser == null) {
73        final long id = self.getLong("id");
74        selfUser = new SelfUserImpl(id, api);
75        api.setSelfUser(selfUser);
76      }
77  
78      if (!api.getUserMap().containsKey(selfUser.getIdLong()))
79        api.getUserMap().put(selfUser.getIdLong(), selfUser);
80  
81      selfUser
82          .setVerified(self.getBoolean("verified"))
83          .setMfaEnabled(self.getBoolean("mfa_enabled"))
84          .setName(self.getString("username"))
85          .setDiscriminator(self.getString("discriminator"))
86          .setAvatarId(self.optString("avatar", null))
87          .setBot(Helpers.optBoolean(self, "bot"));
88  
89      if (this.api.getAccountType() == AccountType.CLIENT) {
90        selfUser
91            .setEmail(self.optString("email", null))
92            .setMobile(Helpers.optBoolean(self, "mobile"))
93            .setNitro(Helpers.optBoolean(self, "premium"))
94            .setPhoneNumber(self.optString("phone", null));
95      }
96  
97      return selfUser;

104   public void createGuildFirstPass(JSONObject guild, Consumer<Guild> secondPassCallback) {
105     final long id = guild.getLong("id");
106     GuildImpl guildObj = ((GuildImpl) api.getGuildMap().get(id));
107     if (guildObj == null) {
108       guildObj = new GuildImpl(api, id);
109       api.getGuildMap().put(id, guildObj);
110     }
111     if (Helpers.optBoolean(guild, "unavailable")) {
112       guildObj.setAvailable(false);
113       // This is used for when GuildCreateHandler receives a guild that is currently unavailable.
114       // During normal READY
115       // loading for bots (which unavailable is always true) the secondPassCallback parameter will
116       // always
117       // be null.
118       if (secondPassCallback != null) secondPassCallback.accept(guildObj);
119       api.getGuildLock().lock(id);
120       return;
121     }
122 
123     // If we make it to here, the Guild is available. This means 1 of 2 things:
124     // Either:
125     // 1) This is Guild provided during READY for a Client account
126     // 2) This is a Guild received from GuildCreateHandler from a GUILD_CREATE event.
127     //      This could be triggered by joining a guild or due to discord finally
128     //      providing us with Guild information about a previously unavailable guild.
129     //      Whether it was unavailable due to Bot READY unavailability or due to an
130     //      outage within discord matters now.
131     //
132     // Either way, we now have enough information to fill in the general information about the
133     // Guild.
134     // This does NOT necessarily mean that we have all information to complete the guild.
135     // For Client accounts, we will also need to use op 12 (GUILD_SYNC) to get all presences of
136     // online users because
137     // discord only provides Online users that we have an open PM channel with or are friends with
138     // for Client accounts.
139     // On larger guilds we will still need to request all users using op 8 (GUILD_MEMBERS_CHUNK).
140     //
141     // The code below takes the information we -do- have and starts to fill in the Guild. It won't
142     // create anything
143     // that might rely on Users that we don't have due to needing the GUILD_MEMBERS_CHUNK
144     // This includes making VoiceStatus and PermissionOverrides
145 
146     guildObj
147         .setAvailable(true)
148         .setIconId(guild.optString("icon", null))
149         .setSplashId(guild.optString("splash", null))
150         .setRegion(guild.getString("region"))
151         .setName(guild.getString("name"))
152         .setAfkTimeout(Guild.Timeout.fromKey(guild.getInt("afk_timeout")))
153         .setVerificationLevel(Guild.VerificationLevel.fromKey(guild.getInt("verification_level")))
154         .setDefaultNotificationLevel(
155             Guild.NotificationLevel.fromKey(guild.getInt("default_message_notifications")))
156         .setRequiredMFALevel(Guild.MFALevel.fromKey(guild.getInt("mfa_level")))
157         .setExplicitContentLevel(
158             Guild.ExplicitContentLevel.fromKey(guild.getInt("explicit_content_filter")));
159 
160     if (guild.isNull("features")) guildObj.setFeatures(Collections.emptySet());
161     else {
162       guildObj.setFeatures(
163           StreamSupport.stream(guild.getJSONArray("features").spliterator(), false)
164               .map(String::valueOf)
165               .collect(Collectors.toSet()));
166     }
167 
168     JSONArray roles = guild.getJSONArray("roles");
169     for (int i = 0; i < roles.length(); i++) {
170       Role role = createRole(roles.getJSONObject(i), guildObj.getIdLong());
171       guildObj.getRolesMap().put(role.getIdLong(), role);
172       if (role.getIdLong() == guildObj.getIdLong()) guildObj.setPublicRole(role);
173     }
174 
175     if (!guild.isNull("emojis")) {
176       JSONArray array = guild.getJSONArray("emojis");
177       TLongObjectMap<Emote> emoteMap = guildObj.getEmoteMap();
178       for (int i = 0; i < array.length(); i++) {
179         JSONObject object = array.getJSONObject(i);
180         if (object.isNull("id")) {
181           LOG.error("Received GUILD_CREATE with an emoji with a null ID. JSON: {}", object);
182           continue;
183         }
184         JSONArray emoteRoles =
185             object.isNull("roles") ? new JSONArray() : object.getJSONArray("roles");
186         final long emoteId = object.getLong("id");
187 
188         EmoteImpl emoteObj = new EmoteImpl(emoteId, guildObj);
189         Set<Role> roleSet = emoteObj.getRoleSet();
190 
191         for (int j = 0; j < emoteRoles.length(); j++)
192           roleSet.add(guildObj.getRoleById(emoteRoles.getString(j)));
193         emoteMap.put(
194             emoteId,
195             emoteObj
196                 .setName(object.optString("name"))
197                 .setAnimated(object.optBoolean("animated"))
198                 .setManaged(Helpers.optBoolean(object, "managed")));
199       }
200     }
201 
202     if (guild.has("members")) {
203       JSONArray members = guild.getJSONArray("members");
204       createGuildMemberPass(guildObj, members);
205     }
206 
207     // This could be null for Client accounts. Will be fixed by GUILD_SYNC
208     Member owner = guildObj.getMemberById(guild.getLong("owner_id"));
209     if (owner != null) guildObj.setOwner(owner);
210 
211     if (guild.has("presences")) {
212       JSONArray presences = guild.getJSONArray("presences");
213       for (int i = 0; i < presences.length(); i++) {
214         JSONObject presence = presences.getJSONObject(i);
215         final long userId = presence.getJSONObject("user").getLong("id");
216         MemberImpl member = (MemberImpl) guildObj.getMembersMap().get(userId);
217 
218         if (member == null)
219           LOG.debug(
220               "Received a ghost presence in GuildFirstPass! UserId: {} Guild: {}",
221               userId,
222               guildObj);
223         else createPresence(member, presence);
224       }
225     }
226 
227     if (guild.has("channels")) {
228       JSONArray channels = guild.getJSONArray("channels");
229 
230       for (int i = 0; i < channels.length(); i++) {
231         JSONObject channel = channels.getJSONObject(i);
232         ChannelType type = ChannelType.fromId(channel.getInt("type"));
233         switch (type) {
234           case TEXT:
235             createTextChannel(channel, guildObj.getIdLong(), false);
236             break;
237           case VOICE:
238             createVoiceChannel(channel, guildObj.getIdLong(), false);
239             break;
240           case CATEGORY:
241             createCategory(channel, guildObj.getIdLong(), false);
242             break;
243           default:
244             LOG.error(
245                 "Received a channel for a guild that isn't a text, voice or category channel. JSON: {}",
246                 channel);
247         }
248       }
249     }
250 
251     if (!guild.isNull("system_channel_id"))
252       guildObj.setSystemChannel(
253           guildObj.getTextChannelsMap().get(guild.getLong("system_channel_id")));
254 
255     if (!guild.isNull("afk_channel_id"))
256       guildObj.setAfkChannel(guildObj.getVoiceChannelsMap().get(guild.getLong("afk_channel_id")));
257 
258     // If the members that we were provided with (and loaded above) were not all of the
259     //  the members in this guild, then we need to request more users from Discord using
260     //  op 9 (GUILD_MEMBERS_CHUNK). To do so, we will cache the guild's JSON so we can properly
261     //  load stuff that relies on Users like Channels, PermissionOverrides and VoiceStatuses
262     //  after we have the rest of the users. We will request the GUILD_MEMBERS_CHUNK information
263     //  which will be sent from discord over the main Websocket and will be handled by
264     //  GuildMemberChunkHandler. After the handler has received all users as determined by the
265     //  value set using `setExpectedGuildMembers`, it will do one of the following:
266     //    1) If this is a Bot account, immediately call EntityBuilder#createGuildSecondPass, thus
267     // finishing
268     //        the Guild object creation process.
269     //    2) If this is a Client account, it will request op 12 (GUILD_SYNC) to make sure we have
270     // all information
271     //        about online users as GUILD_MEMBERS_CHUNK does not include presence information, and
272     // when loading the
273     //        members from GUILD_MEMBERS_CHUNK, we assume they are offline. GUILD_SYNC makes sure
274     // that we mark them
275     //        properly. After GUILD_SYNC is received by GuildSyncHandler, it will call
276     // EntityBuilder#createGuildSecondPass
277     //
278     // If we actually -did- get all of the users needed, then we don't need to Chunk. Furthermore,
279     // we don't need to use GUILD_SYNC because we always get presences with users thus we have all
280     // information
281     // needed to guild the Guild. We will skip
282     if (guild.getJSONArray("members").length() != guild.getInt("member_count")) {
283       cachedGuildJsons.put(id, guild);
284       cachedGuildCallbacks.put(id, secondPassCallback);
285 
286       GuildMembersChunkHandler handler = api.getClient().getHandler("GUILD_MEMBERS_CHUNK");
287       handler.setExpectedGuildMembers(id, guild.getInt("member_count"));
288 
289       // If we are already past READY / RESUME, then chunk at runtime. Otherwise, pass back to the
290       // ReadyHandler
291       // and let it send a burst chunk request.
292       if (api.getClient().isReady()) {
293         if (api.getAccountType() == AccountType.CLIENT) {
294           JSONObject obj =
295               new JSONObject()
296                   .put("op", WebSocketCode.GUILD_SYNC)
297                   .put("guild_id", guildObj.getId());
298           api.getClient().chunkOrSyncRequest(obj);
299         }
300         JSONObject obj =
301             new JSONObject()
302                 .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
303                 .put("d", new JSONObject().put("guild_id", id).put("query", "").put("limit", 0));
304         api.getClient().chunkOrSyncRequest(obj);
305       } else {
306         ReadyHandler readyHandler = api.getClient().getHandler("READY");
307         readyHandler.acknowledgeGuild(
308             guildObj, true, true, api.getAccountType() == AccountType.CLIENT);
309       }
310 
311       api.getGuildLock().lock(id);
312       return;
313     }
314 
315     // As detailed in the comment above, if we've made it this far then we have all member
316     // information needed to
317     // create the Guild. Thus, we fill in the remaining information, unlock the guild, and provide
318     // the guild
319     // to the callback
320     // This should only occur on small user count guilds.
321 
322     JSONArray channels = guild.getJSONArray("channels");
323     createGuildChannelPass(guildObj, channels); // Actually creates PermissionOverrides
324 
325     JSONArray voiceStates = guild.getJSONArray("voice_states");
326     createGuildVoiceStatePass(guildObj, voiceStates);
327 
328     api.getGuildLock().unlock(guildObj.getIdLong());
329     if (secondPassCallback != null) secondPassCallback.accept(guildObj);
330   }

332   public void createGuildSecondPass(long guildId, List<JSONArray> memberChunks) {
333     JSONObject guildJson = cachedGuildJsons.remove(guildId);
334     Consumer<Guild> secondPassCallback = cachedGuildCallbacks.remove(guildId);
335     GuildImpl guildObj = (GuildImpl) api.getGuildMap().get(guildId);
336 
337     if (guildObj == null)
338       throw new IllegalStateException(
339           "Attempted to perform a second pass on an unknown Guild. Guild not in JDA "
340               + "mapping. GuildId: "
341               + guildId);
342     if (guildJson == null)
343       throw new IllegalStateException(
344           "Attempted to perform a second pass on an unknown Guild. No cached Guild "
345               + "for second pass. GuildId: "
346               + guildId);
347     if (secondPassCallback == null)
348       throw new IllegalArgumentException("No callback provided for the second pass on the Guild!");
349 
350     for (JSONArray chunk : memberChunks) createGuildMemberPass(guildObj, chunk);
351 
352     Member owner = guildObj.getMemberById(guildJson.getLong("owner_id"));
353     if (owner != null) guildObj.setOwner(owner);
354 
355     if (guildObj.getOwner() == null)
356       LOG.error(
357           "Never set the Owner of the Guild: {} because we don't have the owner User object! How?!",
358           guildObj.getId());
359 
360     JSONArray channels = guildJson.getJSONArray("channels");
361     createGuildChannelPass(guildObj, channels);
362 
363     JSONArray voiceStates = guildJson.getJSONArray("voice_states");
364     createGuildVoiceStatePass(guildObj, voiceStates);
365 
366     secondPassCallback.accept(guildObj);
367     api.getGuildLock().unlock(guildId);
368   }

383       else this.createPresence(member, presenceJson);
384     }
385   }

394   private void createGuildChannelPass(GuildImpl guildObj, JSONArray channels) {
395     for (int i = 0; i < channels.length(); i++) {
396       JSONObject channel = channels.getJSONObject(i);
397       ChannelType type = ChannelType.fromId(channel.getInt("type"));
398       Channel channelObj = null;
399       switch (type) {
400         case TEXT:
401           channelObj = api.getTextChannelById(channel.getLong("id"));
402           break;
403         case VOICE:
404           channelObj = api.getVoiceChannelById(channel.getLong("id"));
405           break;
406         case CATEGORY:
407           channelObj = api.getCategoryMap().get(channel.getLong("id"));
408           break;
409         default:
410           LOG.error(
411               "Received a channel for a guild that isn't a text, voice or category channel (ChannelPass). JSON: {}",
412               channel);
413       }
414 
415       if (channelObj != null) {
416         JSONArray permissionOverwrites = channel.getJSONArray("permission_overwrites");
417         createOverridesPass((AbstractChannelImpl<?>) channelObj, permissionOverwrites);
418       } else {
419         LOG.error(
420             "Got permission_override for unknown channel with id: {}", channel.getString("id"));
421       }
422     }
423   }

425   public void createGuildVoiceStatePass(GuildImpl guildObj, JSONArray voiceStates) {
426     for (int i = 0; i < voiceStates.length(); i++) {
427       JSONObject voiceStateJson = voiceStates.getJSONObject(i);
428       final long userId = voiceStateJson.getLong("user_id");
429       Member member = guildObj.getMembersMap().get(userId);
430       if (member == null) {
431         LOG.error(
432             "Received a VoiceState for a unknown Member! GuildId: "
433                 + guildObj.getId()
434                 + " MemberId: "
435                 + voiceStateJson.getString("user_id"));
436         continue;
437       }
438 
439       final long channelId = voiceStateJson.getLong("channel_id");
440       VoiceChannelImpl voiceChannel =
441           (VoiceChannelImpl) guildObj.getVoiceChannelsMap().get(channelId);
442       if (voiceChannel != null)
443         voiceChannel.getConnectedMembersMap().put(member.getUser().getIdLong(), member);
444       else
445         LOG.error(
446             "Received a GuildVoiceState with a channel ID for a non-existent channel! ChannelId: {} GuildId: {} UserId: {}",
447             channelId,
448             guildObj.getId(),
449             userId);
450 
451       // VoiceState is considered volatile so we don't expect anything to actually exist
452       GuildVoiceStateImpl voiceState = (GuildVoiceStateImpl) member.getVoiceState();
453       voiceState
454           .setSelfMuted(Helpers.optBoolean(voiceStateJson, "self_mute"))
455           .setSelfDeafened(Helpers.optBoolean(voiceStateJson, "self_deaf"))
456           .setGuildMuted(Helpers.optBoolean(voiceStateJson, "mute"))
457           .setGuildDeafened(Helpers.optBoolean(voiceStateJson, "deaf"))
458           .setSuppressed(Helpers.optBoolean(voiceStateJson, "suppress"))
459           .setSessionId(voiceStateJson.optString("session_id"))
460           .setConnectedChannel(voiceChannel);
461     }
462   }

472   private User createUser(JSONObject user, boolean fake, boolean modifyCache) {
473     final long id = user.getLong("id");
474     UserImpl userObj;
475 
476     userObj = (UserImpl) api.getUserMap().get(id);
477     if (userObj == null) {
478       userObj = (UserImpl) api.getFakeUserMap().get(id);
479       if (userObj != null) {
480         if (!fake && modifyCache) {
481           api.getFakeUserMap().remove(id);
482           userObj.setFake(false);
483           api.getUserMap().put(userObj.getIdLong(), userObj);
484           if (userObj.hasPrivateChannel()) {
485             PrivateChannelImpl priv = (PrivateChannelImpl) userObj.getPrivateChannel();
486             priv.setFake(false);
487             api.getFakePrivateChannelMap().remove(priv.getIdLong());
488             api.getPrivateChannelMap().put(priv.getIdLong(), priv);
489           }
490         }
491       } else {
492         userObj = new UserImpl(id, api).setFake(fake);
493         if (modifyCache) {
494           if (fake) api.getFakeUserMap().put(id, userObj);
495           else api.getUserMap().put(id, userObj);
496         }
497       }
498     }
499 
500     return userObj
501         .setName(user.getString("username"))
502         .setDiscriminator(user.get("discriminator").toString())
503         .setAvatarId(user.optString("avatar", null))
504         .setBot(Helpers.optBoolean(user, "bot"));
505   }

507   public Member createMember(GuildImpl guild, JSONObject memberJson) {
508     User user = createUser(memberJson.getJSONObject("user"));
509     MemberImpl member = (MemberImpl) guild.getMember(user);
510     if (member == null) {
511       member = new MemberImpl(guild, user);
512       guild.getMembersMap().put(user.getIdLong(), member);
513     }
514 
515     ((GuildVoiceStateImpl) member.getVoiceState())
516         .setGuildMuted(memberJson.getBoolean("mute"))
517         .setGuildDeafened(memberJson.getBoolean("deaf"));
518 
519     member
520         .setJoinDate(OffsetDateTime.parse(memberJson.getString("joined_at")))
521         .setNickname(memberJson.optString("nick", null));
522 
523     JSONArray rolesJson = memberJson.getJSONArray("roles");
524     for (int k = 0; k < rolesJson.length(); k++) {
525       final long roleId = rolesJson.getLong(k);
526       Role r = guild.getRolesMap().get(roleId);
527       if (r == null) {
528         LOG.debug(
529             "Received a Member with an unknown Role. MemberId: {} GuildId: {} roleId: {}",
530             member.getUser().getId(),
531             guild.getId(),
532             roleId);
533       } else {
534         member.getRoleSet().add(r);
535       }
536     }
537 
538     return member;
539   }

541   // Effectively the same as createFriendPresence
542   public void createPresence(Object memberOrFriend, JSONObject presenceJson) {
543     if (memberOrFriend == null) throw new NullPointerException("Provided memberOrFriend was null!");
544 
545     JSONObject gameJson = presenceJson.isNull("game") ? null : presenceJson.getJSONObject("game");
546     OnlineStatus onlineStatus = OnlineStatus.fromKey(presenceJson.getString("status"));
547     Game game = null;
548     boolean parsedGame = false;
549 
550     if (gameJson != null && !gameJson.isNull("name")) {
551       try {
552         game = createGame(gameJson);
553         parsedGame = true;
554       } catch (Exception ex) {
555         String userId;
556         if (memberOrFriend instanceof Member) userId = ((Member) memberOrFriend).getUser().getId();
557         else if (memberOrFriend instanceof Friend)
558           userId = ((Friend) memberOrFriend).getUser().getId();
559         else userId = "unknown";
560         if (LOG.isDebugEnabled())
561           LOG.warn(
562               "Encountered exception trying to parse a presence! UserId: {} JSON: {}",
563               userId,
564               gameJson,
565               ex);
566         else
567           LOG.warn(
568               "Encountered exception trying to parse a presence! UserId: {} Message: {} Enable debug for details",
569               userId,
570               ex.getMessage());
571       }
572     }
573     if (memberOrFriend instanceof Member) {
574       MemberImpl member = (MemberImpl) memberOrFriend;
575       member.setOnlineStatus(onlineStatus);

1104     PermissionOverrideImpl permOverride;
1105     final long id = override.getLong("id");
1106     long allow = override.getLong("allow");
1107     long deny = override.getLong("deny");
1108 
1109     // Throwing NoSuchElementException for common issues with overrides that are not cleared

1113       case "member":
1114         Member member = chan.getGuild().getMemberById(id);

1118                   + chan.getGuild()
1119                   + ", Channel: "
1120                   + chan
1121                   + ", JSON: "
1122                   + override);
1123 
1124         permOverride = (PermissionOverrideImpl) chan.getPermissionOverride(member);
1125         if (permOverride == null) {
1126           permOverride = new PermissionOverrideImpl(chan, member.getUser().getIdLong(), member);
1127           ((AbstractChannelImpl<?>) chan)
1128               .getOverrideMap()
1129               .put(member.getUser().getIdLong(), permOverride);
1130         }
1131         break;
1132       case "role":
1133         Role role = ((GuildImpl) chan.getGuild()).getRolesMap().get(id);
1134         if (role == null)
1135           throw new NoSuchElementException(
1136               "Attempted to create a PermissionOverride for a non-existent role! JSON: "
1137                   + override);
1138 
1139         permOverride = (PermissionOverrideImpl) chan.getPermissionOverride(role);
1140         if (permOverride == null) {
1141           permOverride = new PermissionOverrideImpl(chan, role.getIdLong(), role);
1142           ((AbstractChannelImpl<?>) chan).getOverrideMap().put(role.getIdLong(), permOverride);
1143         }
1144         break;
1145       default:
1146         throw new IllegalArgumentException(
1147             "Provided with an unknown PermissionOverride type! JSON: " + override);
1148     }
1149     return permOverride.setAllow(allow).setDeny(deny);
1150   }

1396   public AuditLogChange createAuditLogChange(JSONObject change) {
1397     final String key = change.getString("key");
1398     Object oldValue = change.isNull("old_value") ? null : change.get("old_value");
1399     Object newValue = change.isNull("new_value") ? null : change.get("new_value");
1400 
1401     // Don't confront users with JSON
1402     if (oldValue instanceof JSONArray || newValue instanceof JSONArray) {
1403       oldValue = oldValue instanceof JSONArray ? ((JSONArray) oldValue).toList() : oldValue;
1404       newValue = newValue instanceof JSONArray ? ((JSONArray) newValue).toList() : newValue;
1405     } else if (oldValue instanceof JSONObject || newValue instanceof JSONObject) {
1406       oldValue = oldValue instanceof JSONObject ? ((JSONObject) oldValue).toMap() : oldValue;
1407       newValue = newValue instanceof JSONObject ? ((JSONObject) newValue).toMap() : newValue;
1408     }
1409 
1410     return new AuditLogChange(oldValue, newValue, key);
1411   }

1419   private <T> List<T> map(JSONObject jsonObject, String key, Function<JSONObject, T> convert) {
1420     if (jsonObject.isNull(key)) return Collections.emptyList();
1421 
1422     final JSONArray arr = jsonObject.getJSONArray(key);
1423     final List<T> mappedObjects = new ArrayList<>(arr.length());
1424     for (int i = 0; i < arr.length(); i++) {
1425       JSONObject obj = arr.getJSONObject(i);
1426       mappedObjects.add(convert.apply(obj));
1427     }
1428 
1429     return mappedObjects;
1430   }
