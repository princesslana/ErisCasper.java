17  package net.dv8tion.jda.core.entities;
18  
19  import net.dv8tion.jda.core.JDA;
20  import net.dv8tion.jda.core.managers.WebhookManager;
21  import net.dv8tion.jda.core.managers.WebhookManagerUpdatable;
22  import net.dv8tion.jda.core.requests.restaction.AuditableRestAction;
23  import net.dv8tion.jda.webhook.WebhookClientBuilder;
24  
25  /**
26   * An object representing Webhooks in Discord
27   *
28   * @since 3.0
29   * @author Florian Spieß
30   */
31  public interface Webhook extends ISnowflake {
32  
33    /**
34     * The JDA instance of this Webhook.
35     *
36     * @return The current JDA instance of this Webhook
37     */
38    JDA getJDA();
39  
40    /**
41     * The {@link net.dv8tion.jda.core.entities.Guild Guild} instance for this Webhook. <br>
42     * This is a shortcut for <code>{@link #getChannel()}.getGuild()</code>.
43     *
44     * @return The current Guild of this Webhook
45     */
46    Guild getGuild();
47  
48    /**
49     * The {@link net.dv8tion.jda.core.entities.TextChannel TextChannel} instance this Webhook is
50     * attached to.
51     *
52     * @return The current TextChannel of this Webhook
53     */
54    TextChannel getChannel();
55  
56    /**
57     * The owner of this Webhook.
58     *
59     * @return A {@link net.dv8tion.jda.core.entities.Member Member} instance representing the owner
60     *     of this Webhook
61     */
62    Member getOwner();
63  
64    /**
65     * The default User for this Webhook.
66     *
67     * <p>The {@link net.dv8tion.jda.core.entities.User User} returned is always {@code fake}. <br>
68     * This User is used for all messages posted to the Webhook route (found in {@link #getUrl()}), it
69     * holds the default references for the message authors of messages by this Webhook.
70     *
71     * <p>When {@code POST}ing to a Webhook route the name/avatar of this default user can be
72     * overridden.
73     *
74     * @return A fake {@link net.dv8tion.jda.core.entities.User User} instance representing the
75     *     default webhook user.
76     * @see <a href="https://discordapp.com/developers/docs/resources/webhook#execute-webhook">Execute
77     *     Webhook Docs</a>
78     */
79    User getDefaultUser();
80  
81    /**
82     * The name of this Webhook. <br>
83     * This will be displayed by default as the author name of every message by this Webhook.
84     *
85     * <p>This is a shortcut for <code>{@link #getDefaultUser()}.getName()</code>.
86     *
87     * @return The name of this Webhook
88     */
89    String getName();
90  
91    /**
92     * The execute token for this Webhook. <br>
93     * This can be used to modify/delete/execute this Webhook.
94     *
95     * @return The execute token for this Webhook
96     */
97    String getToken();
98  
99    /**
100    * The {@code POST} route for this Webhook. <br>
101    * This contains the {@link #getToken() token} and {@link #getId() id} of this Webhook.
102    *
103    * <p>The route returned by this method does not need permission checks to be executed. <br>
104    * It is implied that Webhook messages always have all permissions including {@link
105    * net.dv8tion.jda.core.Permission#MESSAGE_MENTION_EVERYONE mentioning everyone}.
106    *
107    * <p>Webhook executions are limited with 5 requests per second. The response contains rate limit
108    * headers that should be handled by execution frameworks. (<a
109    * href="https://discordapp.com/developers/docs/topics/rate-limits">Learn More</a>)
110    *
111    * @return The execution route for this Webhook.
112    */
113   String getUrl();
114 
115   /**
116    * Deletes this Webhook.
117    *
118    * @return {@link net.dv8tion.jda.core.requests.restaction.AuditableRestAction
119    *     AuditableRestAction} <br>
120    *     The rest action to delete this Webhook.
121    */
122   AuditableRestAction<Void> delete();
123 
124   /**
125    * The {@link net.dv8tion.jda.core.managers.WebhookManager Manager} for this Webhook. <br>
126    * This Manager <b>does not</b> require to update, it provides set methods to atomically modify
127    * fields of this Webhook.
128    *
129    * @return An instance of {@link net.dv8tion.jda.core.managers.WebhookManager WebhookManager} for
130    *     this Webhook
131    */
132   WebhookManager getManager();
133 
134   /**
135    * The {@link net.dv8tion.jda.core.managers.WebhookManager Manager} for this Webhook. <br>
136    * This Manager <b>does</b> require to update, it provides get methods to retrieve {@link
137    * net.dv8tion.jda.core.managers.fields.WebhookField fields} of this Webhook which can be modified
138    * and updated.
139    *
140    * @return An instance of {@link net.dv8tion.jda.core.managers.WebhookManagerUpdatable
141    *     WebhookManagerUpdatable} for this Webhook
142    */
143   WebhookManagerUpdatable getManagerUpdatable();
144 
145   /**
146    * Creates a new {@link net.dv8tion.jda.webhook.WebhookClientBuilder WebhookClientBuilder}
147    * instance for this Webhook instance.
148    *
149    * <p><b><u>Remember to close the WebhookClient once you don't need it anymore to free
150    * resources!</u></b>
151    *
152    * @return The new WebhookClientBuilder
153    */
154   WebhookClientBuilder newClient();
155 }
