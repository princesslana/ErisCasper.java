114       boolean autoReconnect,

149   public int login(String gatewayUrl, ShardInfo shardInfo) throws LoginException {
150     this.gatewayUrl = gatewayUrl;
151     this.shardInfo = shardInfo;
152 
153     setStatus(Status.LOGGING_IN);
154     if (token == null || token.isEmpty())
155       throw new LoginException("Provided token was null or empty!");
156 
157     if (contextMap != null && shardInfo != null) {
158       contextMap.put("jda.shard", shardInfo.getShardString());
159       contextMap.put("jda.shard.id", String.valueOf(shardInfo.getShardId()));
160       contextMap.put("jda.shard.total", String.valueOf(shardInfo.getShardTotal()));
161     }
162     verifyToken();
163     LOG.info("Login Successful!");
164 
165     client = new WebSocketClient(this);
166 
167     if (shutdownHook != null) Runtime.getRuntime().addShutdownHook(shutdownHook);
168 
169     return shardInfo == null ? -1 : shardInfo.getShardTotal();
170   }

205   // @param alreadyFailed If has already been a failed attempt with the current configuration
206   public void verifyToken(boolean alreadyFailed) throws LoginException {
207 
208     RestAction<JSONObject> login =
209         new RestAction<JSONObject>(this, Route.Self.GET_SELF.compile()) {
210           @Override
211           protected void handleResponse(Response response, Request<JSONObject> request) {
212             if (response.isOk()) request.onSuccess(response.getObject());
213             else if (response.isRateLimit())
214               request.onFailure(new RateLimitedException(request.getRoute(), response.retryAfter));
215             else if (response.code == 401) request.onSuccess(null);
216             else
217               request.onFailure(
218                   new LoginException(
219                       "When verifying the authenticity of the provided token, Discord returned an unknown response:\n"
220                           + response.toString()));
221           }
222         };
223 
224     JSONObject userResponse;
225 
226     if (!alreadyFailed) {
227       userResponse = checkToken(login);
228       if (userResponse != null) {
229         verifyAccountType(userResponse);
230         return;
231       }
232     }
233 
234     // If we received a null return for userResponse, then that means we hit a 401.
235     // 401 occurs when we attempt to access the users/@me endpoint with the wrong token prefix.
236     // e.g: If we use a Client token and prefix it with "Bot ", or use a bot token and don't prefix
237     // it.
238     // It also occurs when we attempt to access the endpoint with an invalid token.
239     // The code below already knows that something is wrong with the token. We want to determine if
240     // it is invalid
241     // or if the developer attempted to login with a token using the wrong AccountType.
242 
243     // If we attempted to login as a Bot, remove the "Bot " prefix and set the Requester to be a
244     // client.
245     if (getAccountType() == AccountType.BOT) {
246       token = token.substring("Bot ".length());
247       requester = new Requester(this, AccountType.CLIENT);
248     } else // If we attempted to login as a Client, prepend the "Bot " prefix and set the Requester
249     // to be a Bot
250     {
251       token = "Bot " + token;
252       requester = new Requester(this, AccountType.BOT);
253     }
254 
255     userResponse = checkToken(login);
256 
257     // If the response isn't null (thus it didn't 401) send it to the secondary verify method to
258     // determine
259     // which account type the developer wrongly attempted to login as
260     if (userResponse != null) verifyAccountType(userResponse);
261     else // We 401'd again. This is an invalid token
262     throw new LoginException("The provided token is invalid!");
263   }
264 
265   private void verifyAccountType(JSONObject userResponse) {
266     if (getAccountType() == AccountType.BOT) {
267       if (!userResponse.has("bot") || !userResponse.getBoolean("bot"))
268         throw new AccountTypeException(
269             AccountType.BOT, "Attempted to login as a BOT with a CLIENT token!");
270     } else {
271       if (userResponse.has("bot") && userResponse.getBoolean("bot"))
272         throw new AccountTypeException(
273             AccountType.CLIENT, "Attempted to login as a CLIENT with a BOT token!");
274     }
275   }
276 
277   private JSONObject checkToken(RestAction<JSONObject> login) throws LoginException {
278     JSONObject userResponse;
279     try {
280       userResponse = login.complete();
281     } catch (RuntimeException e) {
282       // We check if the LoginException is masked inside of a ExecutionException which is masked
283       // inside of the RuntimeException
284       Throwable ex = e.getCause() instanceof ExecutionException ? e.getCause().getCause() : null;
285       if (ex instanceof LoginException) throw new LoginException(ex.getMessage());
286       else throw e;
287     }
288     return userResponse;
289   }

364   @Override
365   public RestAction<User> retrieveUserById(long id) {
366     AccountTypeException.check(accountType, AccountType.BOT);
367 
368     // check cache
369     User user = this.getUserById(id);
370     if (user != null) return new RestAction.EmptyRestAction<>(this, user);
371 
372     Route.CompiledRoute route = Route.Users.GET_USER.compile(Long.toUnsignedString(id));
373     return new RestAction<User>(this, route) {
374       @Override
375       protected void handleResponse(Response response, Request<User> request) {
376         if (!response.isOk()) {
377           request.onFailure(response);
378           return;
379         }
380         JSONObject user = response.getObject();
381         request.onSuccess(getEntityBuilder().createFakeUser(user, false));
382       }
383     };
384   }

443   @Override
444   public void shutdown() {
445     if (status == Status.SHUTDOWN || status == Status.SHUTTING_DOWN) return;
446 
447     setStatus(Status.SHUTTING_DOWN);
448     audioManagers.forEach(AudioManager::closeAudioConnection);
449     audioManagers.clear();
450 
451     if (audioKeepAlivePool != null) audioKeepAlivePool.shutdownNow();
452 
453     getClient().shutdown();
454 
455     final long time = 5L;
456     final TimeUnit unit = TimeUnit.SECONDS;
457     getRequester().shutdown(time, unit);
458     pool.setKeepAliveTime(time, unit);
459     pool.allowCoreThreadTimeOut(true);
460 
461     if (shutdownHook != null) {
462       try {
463         Runtime.getRuntime().removeShutdownHook(shutdownHook);
464       } catch (Exception ignored) {
465       }
466     }
467 
468     setStatus(Status.SHUTDOWN);
469   }

541   public GuildAction createGuild(String name) {
542     switch (accountType) {
543       case BOT:
544         if (guildCache.size() >= 10)
545           throw new IllegalStateException(
546               "Cannot create a Guild with a Bot in more than 10 guilds!");
547         break;
548       case CLIENT:
549         if (guildCache.size() >= 100)
550           throw new IllegalStateException(
551               "Cannot be in more than 100 guilds with AccountType.CLIENT!");
552     }
553     return new GuildAction(this, name);
554   }

641   private class JDAThreadFactory implements ThreadFactory {
642     @Override
643     public Thread newThread(Runnable r) {
644       final Thread thread = new Thread(r, "JDA-Thread " + getIdentifierString());
645       thread.setDaemon(true);
646       return thread;
647     }

650   public ScheduledThreadPoolExecutor getAudioKeepAlivePool() {
651     ScheduledThreadPoolExecutor akap = audioKeepAlivePool;
652     if (akap == null) {
653       synchronized (akapLock) {
654         akap = audioKeepAlivePool;
655         if (akap == null)
656           akap =
657               audioKeepAlivePool =
658                   new ScheduledThreadPoolExecutor(
659                       1, new AudioWebSocket.KeepAliveThreadFactory(this));
660       }
661     }
662     return akap;
663   }
