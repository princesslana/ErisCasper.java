126   @Override
127   public RestAction<String> getVanityUrl() {
128     if (!isAvailable()) throw new GuildUnavailableException();
129     if (!getSelfMember().hasPermission(Permission.MANAGE_SERVER))
130       throw new InsufficientPermissionException(Permission.MANAGE_SERVER);
131     if (!getFeatures().contains("VANITY_URL"))
132       throw new IllegalStateException("This guild doesn't have a vanity url");
133 
134     Route.CompiledRoute route = Route.Guilds.GET_VANITY_URL.compile(getId());
135 
136     return new RestAction<String>(api, route) {
137       @Override
138       protected void handleResponse(Response response, Request<String> request) {
139         if (!response.isOk()) {
140           request.onFailure(response);
141           return;
142         }
143 
144         request.onSuccess(response.getObject().getString("code"));
145       }
146     };
147   }

159   @Override
160   public RestAction<List<Webhook>> getWebhooks() {
161     if (!getSelfMember().hasPermission(Permission.MANAGE_WEBHOOKS))
162       throw new InsufficientPermissionException(Permission.MANAGE_WEBHOOKS);
163 
164     Route.CompiledRoute route = Route.Guilds.GET_WEBHOOKS.compile(getId());
165 
166     return new RestAction<List<Webhook>>(api, route) {
167       @Override
168       protected void handleResponse(Response response, Request<List<Webhook>> request) {
169         if (!response.isOk()) {
170           request.onFailure(response);
171           return;
172         }
173 
174         JSONArray array = response.getArray();
175         List<Webhook> webhooks = new ArrayList<>(array.length());
176         EntityBuilder builder = api.getEntityBuilder();
177 
178         for (Object object : array) {
179           try {
180             webhooks.add(builder.createWebhook((JSONObject) object));
181           } catch (JSONException | NullPointerException e) {
182             JDAImpl.LOG.error("Error creating webhook from json", e);
183           }
184         }
185 
186         request.onSuccess(Collections.unmodifiableList(webhooks));
187       }
188     };
189   }

251   @Override
252   public RestAction<List<Ban>> getBanList() {
253     if (!isAvailable()) throw new GuildUnavailableException();
254     if (!getSelfMember().hasPermission(Permission.BAN_MEMBERS))
255       throw new InsufficientPermissionException(Permission.BAN_MEMBERS);
256 
257     Route.CompiledRoute route = Route.Guilds.GET_BANS.compile(getId());
258     return new RestAction<List<Ban>>(getJDA(), route) {
259       @Override
260       protected void handleResponse(Response response, Request<List<Ban>> request) {
261         if (!response.isOk()) {
262           request.onFailure(response);
263           return;
264         }
265 
266         EntityBuilder builder = api.getEntityBuilder();
267         List<Ban> bans = new LinkedList<>();
268         JSONArray bannedArr = response.getArray();
269 
270         for (int i = 0; i < bannedArr.length(); i++) {
271           final JSONObject object = bannedArr.getJSONObject(i);
272           JSONObject user = object.getJSONObject("user");
273           bans.add(new Ban(builder.createFakeUser(user, false), object.optString("reason", null)));
274         }
275         request.onSuccess(Collections.unmodifiableList(bans));
276       }
277     };
278   }

280   @Override
281   public RestAction<Integer> getPrunableMemberCount(int days) {
282     if (!isAvailable()) throw new GuildUnavailableException();
283     if (!getSelfMember().hasPermission(Permission.KICK_MEMBERS))
284       throw new InsufficientPermissionException(Permission.KICK_MEMBERS);
285 
286     if (days < 1) throw new IllegalArgumentException("Days amount must be at minimum 1 day.");
287 
288     Route.CompiledRoute route =
289         Route.Guilds.PRUNABLE_COUNT
290             .compile(getId())
291             .withQueryParams("days", Integer.toString(days));
292     return new RestAction<Integer>(getJDA(), route) {
293       @Override
294       protected void handleResponse(Response response, Request<Integer> request) {
295         if (response.isOk()) request.onSuccess(response.getObject().getInt("pruned"));
296         else request.onFailure(response);
297       }
298     };
299   }

364   @Override
365   public RestAction<Void> leave() {
366     if (owner.equals(getSelfMember()))
367       throw new IllegalStateException(
368           "Cannot leave a guild that you are the owner of! Transfer guild ownership first!");
369 
370     Route.CompiledRoute route = Route.Self.LEAVE_GUILD.compile(getId());
371     return new RestAction<Void>(api, route) {
372       @Override
373       protected void handleResponse(Response response, Request<Void> request) {
374         if (response.isOk()) request.onSuccess(null);
375         else request.onFailure(response);
376       }
377     };
378   }

389   @Override
390   public RestAction<Void> delete(String mfaCode) {
391     if (!owner.equals(getSelfMember()))
392       throw new PermissionException("Cannot delete a guild that you do not own!");
393 
394     JSONObject mfaBody = null;
395     if (api.getSelfUser().isMfaEnabled()) {
396       Checks.notEmpty(mfaCode, "Provided MultiFactor Auth code");
397       mfaBody = new JSONObject().put("code", mfaCode);
398     }
399 
400     Route.CompiledRoute route = Route.Guilds.DELETE_GUILD.compile(getId());
401     return new RestAction<Void>(api, route, mfaBody) {
402       @Override
403       protected void handleResponse(Response response, Request<Void> request) {
404         if (response.isOk()) request.onSuccess(null);
405         else request.onFailure(response);
406       }
407     };
408   }

410   @Override
411   public AudioManager getAudioManager() {
412     if (!api.isAudioEnabled())
413       throw new IllegalStateException(
414           "Audio is disabled. Cannot retrieve an AudioManager while audio is disabled.");
415 
416     final TLongObjectMap<AudioManager> managerMap = api.getAudioManagerMap();
417     AudioManager mng = managerMap.get(id);
418     if (mng == null) {
419       // No previous manager found -> create one
420       synchronized (managerMap) {
421         mng = managerMap.get(id);
422         if (mng == null) {
423           mng = new AudioManagerImpl(this);
424           managerMap.put(id, mng);
425         }
426       }
427     }
428     return mng;
429   }

607   @Override
608   public RestAction<List<Invite>> getInvites() {
609     if (!this.getSelfMember().hasPermission(Permission.MANAGE_SERVER))
610       throw new InsufficientPermissionException(Permission.MANAGE_SERVER);
611 
612     final Route.CompiledRoute route = Route.Invites.GET_GUILD_INVITES.compile(getId());
613 
614     return new RestAction<List<Invite>>(api, route) {
615       @Override
616       protected void handleResponse(final Response response, final Request<List<Invite>> request) {
617         if (response.isOk()) {
618           EntityBuilder entityBuilder = this.api.getEntityBuilder();
619           JSONArray array = response.getArray();
620           List<Invite> invites = new ArrayList<>(array.length());
621           for (int i = 0; i < array.length(); i++)
622             invites.add(entityBuilder.createInvite(array.getJSONObject(i)));
623           request.onSuccess(Collections.unmodifiableList(invites));
624         } else {
625           request.onFailure(response);
626         }
627       }
628     };
